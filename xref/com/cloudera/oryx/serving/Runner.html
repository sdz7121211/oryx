70  /**
71   * <p>This is the runnable class which starts the Serving Layer and its Tomcat-based HTTP server. It is
72   * started with {@link #call()} and can be shut down with {@link #close()}. This implementation is used
73   * both in stand-alone local mode, and in a distributed mode cooperating with a Computation Layer.</p>
74   *
75   * <p>This program instantiates a Tomcat-based HTTP server exposing a REST-style API. It is available via
76   * HTTP, or HTTPS as well if "serving-layer.api.keystore-file" is set. It can also be password
77   * protected by setting "serving-layer.api.user-name" and
78   * "serving-layer.api.password".</p>
79   *
80   * <p>When run in local mode, the Serving Layer instance will compute a model locally and save it as files
81   * under a {@code model/} directory. It will be updated when the model is rebuilt.
82   * If the file is present at startup, it will be read to restore the server state, rather than re-reading
83   * CSV input in the directory and recomputing the model. Thus the file can be saved and restored as a
84   * way of preserving and recalling the server's state of learning.</p>
85   *
86   * <p>Example of running in local mode:</p>
87   *
88   * <p>{@code java -jar oryx-serving-x.y.jar}</p>
89   *
90   * <p>(with an example of JVM tuning flags:)</p>
91   *
92   * <p>{@code java -Xmx1g -XX:NewRatio=12 -XX:+UseParallelOldGC -jar oryx-serving-x.y.jar}</p>
93   *
94   * <p>Finally, some more advanced tuning parameters are available. These are system parameters, set with
95   * {@code -Dproperty=value}.</p>
96   *
97   * @author Sean Owen
98   */
99  public final class Runner implements Callable<Object>, Closeable {

131     this.noSuchBaseDir.deleteOnExit();
132   }

134   public static void main(String[] args) throws Exception {
135     final Runner runner = new Runner();
136     SignalManager.register(new Runnable() {
137       @Override
138       public void run() {
139         runner.close();
140       }
141     }, SignalType.INT, SignalType.TERM);
142     try {
143       runner.call();
144       runner.await();
145     } finally {
146       runner.close();
147     }
148   }

150   @Override
151   public synchronized Object call() throws IOException {
152 
153     if (closed) {
154       return Boolean.FALSE;
155     }
156 
157     MemoryHandler.setSensibleLogFormat();
158     java.util.logging.Logger.getLogger("").addHandler(new MemoryHandler());
159 
160     System.setProperty("org.apache.tomcat.util.buf.UDecoder.ALLOW_ENCODED_SLASH", "true");
161 
162     Tomcat tomcat = new Tomcat();
163     Connector connector = makeConnector();
164     configureTomcat(tomcat, connector);
165     configureEngine(tomcat.getEngine());
166     configureServer(tomcat.getServer());
167     configureHost(tomcat.getHost());
168     Context context = makeContext(tomcat, noSuchBaseDir);
169 
170     AbstractOryxServingInitListener apiListener = chooseAPIListener();
171     context.addApplicationListener(new ApplicationListener(apiListener.getClass().getName(), false));
172 
173     AbstractOryxServingInitListener.addServlet(context, new style_jspx(), "/style.jspx");
174     AbstractOryxServingInitListener.addServlet(context, new index_jspx(), "/index.jspx");
175     AbstractOryxServingInitListener.addServlet(context, new status_jspx(), "/status.jspx");
176     AbstractOryxServingInitListener.addServlet(context, new error_jspx(), "/error.jspx");
177     AbstractOryxServingInitListener.addServlet(context, new LogServlet(), "/log.txt");
178     AbstractOryxServingInitListener.addServlet(context, new ConfigServlet(), "/config.json");
179     AbstractOryxServingInitListener.addServlet(context, apiListener.getApiJSPXFile(), "/test.jspx");
180     apiListener.addServlets(context);
181 
182     try {
183       tomcat.start();
184     } catch (LifecycleException le) {
185       throw new IOException(le);
186     }
187     this.tomcat = tomcat;
188     return Boolean.TRUE;
189   }
190 
191   /**
192    * Blocks and waits until the server shuts down.
193    */
194   void await() {
195     Server server = null;
196     synchronized (this) {
197       if (tomcat != null) {
198         server = tomcat.getServer();
199       }
200     }
201     if (server != null) {
202       server.await();
203     }
204   }
205 
206   @Override
207   public synchronized void close() {
208     if (!closed) {
209       closed = true;
210       if (tomcat != null) {
211         try {
212           tomcat.stop();
213           tomcat.destroy();
214         } catch (LifecycleException le) {
215           log.warn("Unexpected error while stopping", le);
216         }
217         try {
218           IOUtils.deleteRecursively(noSuchBaseDir);
219         } catch (IOException e) {
220           log.warn("Failed to delete {}", noSuchBaseDir);
221         }
222       }
223     }
224   }

226   private void configureTomcat(Tomcat tomcat, Connector connector) {
227     tomcat.setBaseDir(noSuchBaseDir.getAbsolutePath());
228     tomcat.setConnector(connector);
229     tomcat.getService().addConnector(connector);
230   }

232   private void configureEngine(Engine engine) {
233     APISettings apiSettings = APISettings.create(config.getConfig("serving-layer.api"));
234     String userName = apiSettings.getUserName();
235     String password = apiSettings.getPassword();
236     if (userName != null && password != null) {
237       InMemoryRealm realm = new InMemoryRealm();
238       realm.addUser(userName, password);
239       engine.setRealm(realm);
240     }
241   }

243   private static void configureServer(Server server) {
244     server.addLifecycleListener(new JreMemoryLeakPreventionListener());
245     server.addLifecycleListener(new ThreadLocalLeakPreventionListener());
246   }

252   private Connector makeConnector() {
253     Connector connector = new Connector("org.apache.coyote.http11.Http11NioProtocol");
254     APISettings apiSettings = APISettings.create(config.getConfig("serving-layer.api"));
255 
256     File keystoreFile = apiSettings.getKeystoreFile();
257     String keystorePassword = apiSettings.getKeystorePassword();
258     if (keystoreFile == null && keystorePassword == null) {
259       // HTTP connector
260       connector.setPort(apiSettings.getPort());
261       connector.setSecure(false);
262       connector.setScheme("http");
263 
264     } else {
265 
266       // HTTPS connector
267       connector.setPort(apiSettings.getSecurePort());
268       connector.setSecure(true);
269       connector.setScheme("https");
270       connector.setAttribute("SSLEnabled", "true");
271       String protocol = chooseSSLProtocol("TLSv1.1", "TLSv1");
272       if (protocol != null) {
273         connector.setAttribute("sslProtocol", protocol);
274       }
275       if (keystoreFile != null) {
276         connector.setAttribute("keystoreFile", keystoreFile.getAbsoluteFile());
277       }
278       connector.setAttribute("keystorePass", keystorePassword);
279     }
280 
281     // Keep quiet about the server type
282     connector.setXpoweredBy(false);
283     connector.setAttribute("server", "Oryx");
284 
285     // Basic tuning params:
286     connector.setAttribute("maxThreads", 400);
287     connector.setAttribute("acceptCount", 50);
288     //connector.setAttribute("connectionTimeout", 2000);
289     connector.setAttribute("maxKeepAliveRequests", 100);
290 
291     // Avoid running out of ephemeral ports under heavy load?
292     connector.setAttribute("socket.soReuseAddress", true);
293     
294     connector.setMaxPostSize(0);
295     connector.setAttribute("disableUploadTimeout", false);
296 
297     // Allow long URLs
298     connector.setAttribute("maxHttpHeaderSize", 32768);
299 
300     return connector;
301   }

303   private static String chooseSSLProtocol(String... protocols) {
304     for (String protocol : protocols) {
305       try {
306         SSLContext.getInstance(protocol);
307         return protocol;
308       } catch (NoSuchAlgorithmException ignored) {
309         // continue
310       }
311     }
312     return null;
313   }

315   private Context makeContext(Tomcat tomcat, File noSuchBaseDir) throws IOException {
316 
317     File contextPath = new File(noSuchBaseDir, "context");
318     IOUtils.mkdirs(contextPath);
319 
320     APISettings apiSettings = APISettings.create(config.getConfig("serving-layer.api"));
321     String contextPathURIBase = config.getString("serving-layer.api.context-path");
322     if (contextPathURIBase == null || contextPathURIBase.isEmpty() || "/".equals(contextPathURIBase)) {
323       contextPathURIBase = "";
324     }
325     Context context = tomcat.addContext(contextPathURIBase, contextPath.getAbsolutePath());
326 
327     context.setWebappVersion("3.0");
328     context.addWelcomeFile("index.jspx");
329     addErrorPages(context);
330 
331     context.addServletContainerInitializer(new JasperInitializer(), Collections.<Class<?>>emptySet());
332 
333     boolean needHTTPS = apiSettings.isSecure();
334     boolean needAuthentication = apiSettings.getUserName() != null;
335 
336     if (needHTTPS || needAuthentication) {
337 
338       SecurityCollection securityCollection = new SecurityCollection();
339       securityCollection.addPattern("/*");
340       SecurityConstraint securityConstraint = new SecurityConstraint();
341       securityConstraint.addCollection(securityCollection);
342 
343       if (needHTTPS) {
344         securityConstraint.setUserConstraint("CONFIDENTIAL");
345       }
346 
347       if (needAuthentication) {
348 
349         LoginConfig loginConfig = new LoginConfig();
350         loginConfig.setAuthMethod("DIGEST");
351         loginConfig.setRealmName(InMemoryRealm.NAME);
352         context.setLoginConfig(loginConfig);
353 
354         securityConstraint.addAuthRole(InMemoryRealm.AUTH_ROLE);
355 
356         context.addSecurityRole(InMemoryRealm.AUTH_ROLE);
357         DigestAuthenticator authenticator = new DigestAuthenticator();
358         authenticator.setNonceValidity(10 * 1000L); // Shorten from 5 minutes to 10 seconds
359         authenticator.setNonceCacheSize(20000); // Increase from 1000 to 20000
360         context.getPipeline().addValve(authenticator);
361       }
362 
363       context.addConstraint(securityConstraint);
364     }
365 
366     context.setCookies(false);
367 
368     return context;
369   }

371   private AbstractOryxServingInitListener chooseAPIListener() {
372     Class<? extends AbstractOryxServingInitListener> listenerClass = LISTENERS.get(config.getString("model.type"));
373     Preconditions.checkNotNull(listenerClass,
374                                "Unspecified or unsupported model type: %s",
375                                config.getString("model.type"));
376     return ClassUtils.loadInstanceOf(listenerClass);
377   }

379   private static void addErrorPages(Context context) {
380     for (int errorCode : ERROR_PAGE_STATUSES) {
381       ErrorPage errorPage = new ErrorPage();
382       errorPage.setErrorCode(errorCode);
383       errorPage.setLocation("/error.jspx");
384       context.addErrorPage(errorPage);
385     }
386     ErrorPage errorPage = new ErrorPage();
387     errorPage.setExceptionType(Throwable.class.getName());
388     errorPage.setLocation("/error.jspx");
389     context.addErrorPage(errorPage);
390   }
