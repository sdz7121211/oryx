16  package com.cloudera.oryx.serving.web;
17  
18  import com.google.common.base.Charsets;
19  import com.typesafe.config.ConfigFactory;
20  
21  import javax.servlet.http.HttpServlet;
22  import javax.servlet.http.HttpServletRequest;
23  import javax.servlet.http.HttpServletResponse;
24  import java.io.IOException;
25  
26  /**
27   * A simple servlet that returns a JSON representation of the application's global configuration.
28   */
29  public final class ConfigServlet extends HttpServlet {
30    @Override
31    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws IOException {
32      response.setContentType("application/json");
33      response.setCharacterEncoding(Charsets.UTF_8.name());
34      response.getWriter().write(ConfigFactory.load().root().render());
35    }
36  }
