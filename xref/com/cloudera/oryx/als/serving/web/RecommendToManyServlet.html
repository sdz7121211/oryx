33  /**
34   * <p>Responds to a GET request to
35   * {@code /recommendToMany/[userID1](/[userID2]/...)(?howMany=n)(&considerKnownItems=true|false)(&rescorerParams=...)}
36   * and in turn calls
37   * {@link OryxRecommender#recommendToMany(String[], int, boolean, Rescorer)}.
38   * If {@code howMany} is not specified, defaults to
39   * {@link AbstractALSServlet#DEFAULT_HOW_MANY}. If {@code considerKnownItems} is not specified,
40   * it is considered {@code false}.</p>
41   *
42   * <p>Unknown user IDs are ignored, unless all are unknown, in which case a
43   * {@link HttpServletResponse#SC_BAD_REQUEST} status is returned.</p>
44   *
45   * <p>CSV output contains one recommendation per line, and each line is of the form {@code itemID, strength},
46   * like {@code 325, 0.53}. Strength is an opaque indicator of the relative quality of the recommendation.</p>
47   *
48   * @author Sean Owen
49   */
50  public final class RecommendToManyServlet extends AbstractALSServlet {
51    
52    @Override
53    protected void doGet(HttpServletRequest request, HttpServletResponse response) throws IOException {
54  
55      CharSequence pathInfo = request.getPathInfo();
56      if (pathInfo == null) {
57        response.sendError(HttpServletResponse.SC_BAD_REQUEST, "No path");
58        return;
59      }
60      Iterator<String> pathComponents = SLASH.split(pathInfo).iterator();
61      Set<String> userIDSet = Sets.newHashSet();
62      try {
63        while (pathComponents.hasNext()) {
64          userIDSet.add(pathComponents.next());
65        }
66      } catch (NoSuchElementException nsee) {
67        response.sendError(HttpServletResponse.SC_BAD_REQUEST, nsee.toString());
68        return;
69      }
70  
71      String[] userIDs = userIDSet.toArray(new String[userIDSet.size()]);
72      unescapeSlashHack(userIDs);
73  
74      OryxRecommender recommender = getRecommender();
75      RescorerProvider rescorerProvider = getRescorerProvider();
76      try {
77        Rescorer rescorer = rescorerProvider == null ? null :
78            rescorerProvider.getRecommendRescorer(userIDs, recommender, getRescorerParams(request));
79        output(request, response, recommender.recommendToMany(userIDs,
80                                                              getHowMany(request),
81                                                              getConsiderKnownItems(request),
82                                                              rescorer));
83      } catch (NoSuchUserException nsue) {
84        response.sendError(HttpServletResponse.SC_NOT_FOUND, nsue.toString());
85      } catch (NotReadyException nre) {
86        response.sendError(HttpServletResponse.SC_SERVICE_UNAVAILABLE, nre.toString());
87      } catch (IllegalArgumentException iae) {
88        response.sendError(HttpServletResponse.SC_BAD_REQUEST, iae.toString());
89      }
90    }
91  
92  }
