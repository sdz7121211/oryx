16  package com.cloudera.oryx.als.serving;
17  
18  import java.util.Iterator;
19  
20  import com.cloudera.oryx.als.common.NumericIDValue;
21  import com.cloudera.oryx.als.common.rescorer.Rescorer;
22  import com.cloudera.oryx.als.common.StringLongMapping;
23  import com.cloudera.oryx.common.collection.LongFloatMap;
24  
25  import com.google.common.primitives.Doubles;
26  
27  /**
28   * Used by {@link com.cloudera.oryx.als.common.OryxRecommender#mostPopularItems(int)}.
29   *
30   * @author Sean Owen
31   */
32  final class MostPopularItemsIterator implements Iterator<NumericIDValue> {
33  
34    private final NumericIDValue delegate;
35    private final Iterator<LongFloatMap.MapEntry> countsIterator;
36    private final Rescorer rescorer;
37    private final StringLongMapping idMapping;
38  
39    MostPopularItemsIterator(Iterator<LongFloatMap.MapEntry> countsIterator,
40                             Rescorer rescorer,
41                             StringLongMapping idMapping) {
42      delegate = new NumericIDValue();
43      this.countsIterator = countsIterator;
44      this.rescorer = rescorer;
45      this.idMapping = idMapping;
46    }
47  
48    @Override
49    public boolean hasNext() {
50      return countsIterator.hasNext();
51    }
52  
53    @Override
54    public NumericIDValue next() {
55      LongFloatMap.MapEntry entry = countsIterator.next();
56      long id = entry.getKey();
57      double value = entry.getValue();
58      Rescorer theRescorer = rescorer;
59      if (theRescorer != null) {
60        String stringID = idMapping.toString(id);
61        if (theRescorer.isFiltered(stringID)) {
62          return null;
63        }
64        value = (float) theRescorer.rescore(stringID, value);
65        if (!Doubles.isFinite(value)) {
66          return null;
67        }
68      }
69      delegate.set(id, (float) value);
70      return delegate;
71    }
72  
73    /**
74     * @throws UnsupportedOperationException
75     */
76    @Override
77    public void remove() {
78      throw new UnsupportedOperationException();
79    }
80  
81  }
