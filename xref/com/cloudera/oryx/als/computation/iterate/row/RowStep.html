45  /**
46   * @author Sean Owen
47   */
48  public final class RowStep extends IterationStep {
49  
50    private static final Logger log = LoggerFactory.getLogger(RowStep.class);
51  
52    public static final String Y_KEY_KEY = "Y_KEY";
53    public static final String POPULAR_KEY = "POPULAR";
54    public static final String CONVERGENCE_SAMPLING_MODULUS_KEY = "CONVERGENCE_SAMPLING_MODULUS";
55    public static final String MAP_KEY = "MAP";
56  
57    @Override
58    protected MRPipeline createPipeline() throws IOException {
59  
60      IterationState iterationState = getIterationState();
61      String iterationKey = iterationState.getIterationKey();
62      boolean x = iterationState.isComputingX();
63      int lastIteration = iterationState.getIteration() - 1;
64      Store store = Store.get();
65  
66      JobStepConfig config = getConfig();
67      String instanceDir = config.getInstanceDir();
68      int generationID = config.getGenerationID();
69  
70      if (store.exists(Namespaces.getInstanceGenerationPrefix(instanceDir, generationID) + "X/", false)) {
71        // Actually, looks like whole computation of X/Y finished -- just proceed
72        return null;
73      }
74  
75      // Take the opportunity to clean out iteration before last, if computing X
76      if (x) {
77        String lastLastIterationKey =
78            Namespaces.getIterationsPrefix(instanceDir, generationID) + (lastIteration - 1) + '/';
79        if (store.exists(lastLastIterationKey, false)) {
80          log.info("Deleting old iteration data from {}", lastLastIterationKey);
81          store.recursiveDelete(lastLastIterationKey);
82        }
83      }
84  
85      String yKey;
86      if (x) {
87        yKey = Namespaces.getIterationsPrefix(instanceDir, generationID) + lastIteration + "/Y/";
88      } else {
89        yKey = iterationKey + "X/";
90      }
91  
92      String xKey = iterationKey + (x ? "X/" : "Y/");
93      String rKey = Namespaces.getTempPrefix(instanceDir, generationID) + (x ? "userVectors/" : "itemVectors/");
94  
95      if (!validOutputPath(xKey)) {
96        return null;
97      }
98  
99      MRPipeline p = createBasicPipeline(RowReduceFn.class);
100     Configuration conf = p.getConfiguration();
101     conf.set(Y_KEY_KEY, yKey);
102 
103     String tempKey = Namespaces.getTempPrefix(instanceDir, generationID);
104     String popularKey = tempKey + (x ? "popularItemsByUserPartition/" : "popularUsersByItemPartition/");
105     conf.set(POPULAR_KEY, popularKey);
106 
107     String testPrefix = Namespaces.getInstanceGenerationPrefix(instanceDir, generationID) + "test/";
108     conf.set(MAP_KEY, testPrefix);
109 
110     YState yState = new YState(ALSTypes.DENSE_ROW_MATRIX); // Shared Y-Matrix state
111 
112     GroupingOptions opts = groupingOptions();
113     PCollection<MatrixRow> matrix = PTables.asPTable(p.read(input(rKey, ALSTypes.SPARSE_ROW_MATRIX)))
114         .groupByKey(opts)
115         .parallelDo("rowReduce", new RowReduceFn(yState), ALSTypes.DENSE_ROW_MATRIX)
116         .write(output(xKey));
117 
118     if (!x) {
119       // Configure and perform convergence sampling
120       int modulus = chooseConvergenceSamplingModulus(opts);
121       conf.setInt(CONVERGENCE_SAMPLING_MODULUS_KEY, modulus);
122 
123       matrix
124           .parallelDo("asPair", MatrixRow.AS_PAIR, Avros.tableOf(Avros.longs(), ALSTypes.FLOAT_ARRAY))
125           .parallelDo("convergenceSample", new ConvergenceSampleFn(yState), Avros.strings())
126           .write(compressedTextOutput(p.getConfiguration(), iterationKey + "Yconvergence"));
127     }
128 
129     if (x &&
130         ConfigUtils.getDefaultConfig().getDouble("model.test-set-fraction") > 0.0 &&
131         store.exists(testPrefix, false)) {
132       PCollection<Double> aps = matrix
133           .parallelDo("asPair", MatrixRow.AS_PAIR, Avros.tableOf(Avros.longs(), ALSTypes.FLOAT_ARRAY))
134           .parallelDo("computeAP", new ComputeUserAPFn(yState), Avros.doubles());
135       Mean meanAveragePrecision = new Mean();
136       for (double ap : aps.materialize()) {
137         meanAveragePrecision.increment(ap);
138       }
139       File tempMAPFile = File.createTempFile("MAP", ".txt");
140       tempMAPFile.deleteOnExit();
141       Files.write(Double.toString(meanAveragePrecision.getResult()), tempMAPFile, Charsets.UTF_8);
142       store.upload(iterationKey + "MAP", tempMAPFile, false);
143       IOUtils.delete(tempMAPFile);
144     }
145 
146     return p;
147   }
148   
149   private static int chooseConvergenceSamplingModulus(GroupingOptions opts) {
150     // Kind of arbitrary formula, determined empirically.
151     int modulus = RandomUtils.nextTwinPrime(16 * opts.getNumReducers() * opts.getNumReducers());
152     log.info("Using convergence sampling modulus {} to sample about {}% of all user-item pairs for convergence",
153              modulus, 100.0f / modulus / modulus);
154     return modulus;
155   }
156 
157   @Override
158   protected boolean isHighMemoryStep() {
159     return true;
160   }
161 
162   public static void main(String[] args) throws Exception {
163     run(new RowStep(), args);
164   }
165 
166 }
