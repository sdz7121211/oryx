41  /**
42   * Provides utility methods used by the ALS Computation Layer's MapReduce jobs. These are generally methods
43   * to side-load data from HDFS into memory.
44   *
45   * @author Sean Owen
46   */
47  public final class ComputationDataUtils {
48  
49    private static final Logger log = LoggerFactory.getLogger(ComputationDataUtils.class);
50  
51    private static final Pattern PART_FILE_NAME_PATTERN = Pattern.compile("part-r-(\\d+)");
52  
53    private ComputationDataUtils() {
54    }
55    
56    public static LongObjectMap<float[]> loadPartialY(int partition,
57                                                    int numPartitions,
58                                                    String prefix,
59                                                    Configuration conf) throws IOException {
60      LongObjectMap<float[]> result = new LongObjectMap<float[]>();
61      AvroFileSource<MatrixRow> records = new AvroFileSource<MatrixRow>(
62          Namespaces.toPath(prefix),
63          (AvroType<MatrixRow>) ALSTypes.DENSE_ROW_MATRIX);
64      for (MatrixRow record : records.read(conf)) {
65        long id = record.getRowId();
66        if (LongMath.mod(id, numPartitions) == partition) {
67          result.put(id, record.getValues());
68        }
69      }
70      return result;
71    }
72  
73    public static LongSet readExpectedIDsFromPartition(int currentPartition,
74                                                         int numPartitions,
75                                                         String partitionsPrefix,
76                                                         Progressable progressable,
77                                                         Configuration conf) throws IOException {
78  
79      Store store = Store.get();
80      if (!store.exists(partitionsPrefix, false)) {
81        log.info("No IDs in {}, assuming all are allowed", partitionsPrefix);
82        return null;
83      }
84  
85      List<String> partitionFileKeys = store.list(partitionsPrefix, true);
86      Preconditions.checkState(partitionFileKeys.size() == numPartitions,
87                               "Number of partitions doesn't match number of ID files (%s vs %s). " +
88                                   "Was the number of reducers changed?", partitionFileKeys.size(), numPartitions);
89  
90      // Shuffle order that the many reducers read the many files
91      Collections.shuffle(partitionFileKeys);
92  
93      for (String partitionFileKey : partitionFileKeys) {
94        Matcher m = PART_FILE_NAME_PATTERN.matcher(partitionFileKey);
95        Preconditions.checkState(m.find(), "Bad part path file name: {}", partitionFileKey);
96        int partPartition = Integer.parseInt(m.group(1));
97        if (currentPartition == partPartition) {
98          return readExpectedIDs(partitionFileKey, progressable, conf);
99        }
100     }
101     throw new IllegalStateException("No file found for partition " + currentPartition);
102   }
103 
104   private static LongSet readExpectedIDs(String key,
105                                            Progressable progressable,
106                                            Configuration conf) throws IOException {
107     LongSet ids = new LongSet();
108     long count = 0;
109     for (long id : new AvroFileSource<Long>(Namespaces.toPath(key), Avros.longs()).read(conf)) {
110       ids.add(id);
111       if (++count % 10000 == 0) {
112         progressable.progress();
113       }
114     }
115     log.info("Read {} IDs from {}", ids.size(), key);
116     return ids;
117   }
