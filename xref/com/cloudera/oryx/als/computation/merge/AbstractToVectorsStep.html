20  import com.cloudera.oryx.als.common.NumericIDValue;

27  import com.cloudera.oryx.common.servcomp.Namespaces;
28  
29  /**
30   * @author Sean Owen
31   */
32  abstract class AbstractToVectorsStep extends ALSJobStep {
33  
34    @Override
35    protected final MRPipeline createPipeline() throws IOException {
36  
37      JobStepConfig config = getConfig();
38      String instanceDir = config.getInstanceDir();
39      int generationID = config.getGenerationID();
40  
41      String inputKey = Namespaces.getInstanceGenerationPrefix(instanceDir, generationID) + "input/";
42      String outputKey = Namespaces.getTempPrefix(instanceDir, generationID) + getSuffix();
43      if (!validOutputPath(outputKey)) {
44        return null;
45      }
46  
47      MRPipeline p = createBasicPipeline(ToVectorReduceFn.class);
48      getMatrix(p, inputKey)
49          .groupByKey(groupingOptions())
50          .parallelDo("toVectors", new ToVectorReduceFn(), ALSTypes.SPARSE_ROW_MATRIX)
51          .write(output(outputKey));
52      return p;
53   }
54    
55    abstract PTable<Long, NumericIDValue> getMatrix(MRPipeline pipeline, String inputKey);
56    
57    abstract String getSuffix();
58  
59  }
