16  package com.cloudera.oryx.als.computation.merge;
17  
18  import com.cloudera.oryx.als.common.NumericIDValue;
19  import com.cloudera.oryx.common.collection.LongFloatMap;
20  import com.cloudera.oryx.common.collection.LongSet;
21  import com.cloudera.oryx.common.random.RandomManager;
22  import com.cloudera.oryx.common.settings.ConfigUtils;
23  import com.cloudera.oryx.computation.common.fn.OryxReduceDoFn;
24  import com.google.common.base.Preconditions;
25  import com.typesafe.config.Config;
26  import org.apache.commons.math3.random.RandomGenerator;
27  import org.apache.commons.math3.util.FastMath;
28  import org.apache.crunch.Emitter;
29  import org.apache.crunch.Pair;
30  
31  public final class MergeNewOldValuesFn extends OryxReduceDoFn<Pair<Long, Integer>, Iterable<NumericIDValue>,
32      Pair<Long, NumericIDValue>> {
33  
34    static final int BEFORE = 0;
35    static final int AFTER = 1;
36  
37    private boolean doDecay;
38    private float decayFactor;
39    private float zeroThreshold;
40    private Long previousUserID;
41    private LongFloatMap previousUserPrefs;
42    private double testSetFraction;
43    private RandomGenerator random;
44  
45    @Override
46    public void initialize() {
47      super.initialize();
48      Config config = ConfigUtils.getDefaultConfig();
49      decayFactor = (float) config.getDouble("model.decay.factor");
50      Preconditions.checkArgument(decayFactor > 0.0f && decayFactor <= 1.0f,
51                                  "Decay factor must be in (0,1]: %s", zeroThreshold);
52      zeroThreshold = (float) config.getDouble("model.decay.zeroThreshold");
53      Preconditions.checkArgument(zeroThreshold >= 0.0f,
54                                  "Zero threshold must be nonnegative: %s", zeroThreshold);
55      doDecay = decayFactor < 1.0f;
56  
57      testSetFraction = config.getDouble("model.test-set-fraction");
58      Preconditions.checkArgument(testSetFraction >= 0.0 && testSetFraction <= 1.0);
59      if (testSetFraction > 0.0) {
60        random = RandomManager.getRandom();
61      }
62    }
63  
64    @Override
65    public void process(Pair<Pair<Long, Integer>, Iterable<NumericIDValue>> input,
66                        Emitter<Pair<Long, NumericIDValue>> emitter) {
67      Pair<Long, Integer> key = input.first();
68      long currentUserID = key.first();
69  
70      if (key.second() == BEFORE) {
71  
72        // Last old data had no match, just output it
73        if (previousUserPrefs != null) {
74          Preconditions.checkNotNull(previousUserID);
75          output(previousUserID, previousUserPrefs, null, null, emitter);
76          previousUserPrefs = null;
77          previousUserID = null;
78        }
79  
80        LongFloatMap oldPrefs = new LongFloatMap();
81        for (NumericIDValue itemPref : input.second()) {
82          float oldPrefValue = itemPref.getValue();
83          Preconditions.checkState(!Float.isNaN(oldPrefValue), "No prior pref value?");
84          // Apply decay factor here, if applicable:
85          oldPrefs.increment(itemPref.getID(), doDecay ? oldPrefValue * decayFactor : oldPrefValue);
86        }
87  
88        previousUserPrefs = oldPrefs;
89        previousUserID = currentUserID;
90  
91      } else {
92        // Last old data had no match, just output it
93        if (previousUserPrefs != null && currentUserID != previousUserID) {
94          Preconditions.checkNotNull(previousUserID);
95          output(previousUserID, previousUserPrefs, null, null, emitter);
96          previousUserPrefs = null;
97          previousUserID = null;
98        }
99  
100       LongFloatMap newPrefs = new LongFloatMap();
101       LongSet removedItemIDs = new LongSet();
102       LongFloatMap testSetPrefs = new LongFloatMap();
103 
104       for (NumericIDValue itemPref : input.second()) {
105         long itemID = itemPref.getID();
106         float newPrefValue = itemPref.getValue();
107 
108         // First: do we want to save off this data as test set data?
109         if (testSetFraction > 0.0 && random.nextDouble() < testSetFraction) {
110           testSetPrefs.increment(itemID, newPrefValue);
111         } else {
112           // Normal handling
113           if (Float.isNaN(newPrefValue)) {
114             removedItemIDs.add(itemID);
115           } else {
116             newPrefs.increment(itemID, newPrefValue);
117           }
118         }
119       }
120 
121       output(currentUserID, previousUserPrefs, newPrefs, removedItemIDs, emitter);
122       // TODO other emitter
123       output(currentUserID, null, testSetPrefs, null, emitter);
124 
125       previousUserPrefs = null;
126       previousUserID = null;
127     }
128   }
129 
130   @Override
131   public void cleanup(Emitter<Pair<Long, NumericIDValue>> emitter) {
132     if (previousUserPrefs != null) {
133       Preconditions.checkNotNull(previousUserID);
134       output(previousUserID, previousUserPrefs, null, null, emitter);
135     }
136     super.cleanup(emitter);
137   }
138 
139   private void output(long userID,
140                       LongFloatMap oldPrefs,
141                       LongFloatMap newPrefs,
142                       LongSet removedItemIDs,
143                       Emitter<Pair<Long, NumericIDValue>> emitter) {
144     // Old prefs may be null when there is no previous generation, for example, or the user is new.
145     // First, write out existing prefs, possibly updated by new values
146     if (oldPrefs != null && !oldPrefs.isEmpty()) {
147       for (LongFloatMap.MapEntry entry : oldPrefs.entrySet()) {
148         long itemID = entry.getKey();
149         float oldPrefValue = entry.getValue();
150         Preconditions.checkState(!Float.isNaN(oldPrefValue), "No prior pref value?");
151 
152         // May be NaN if no new data at all, or new data has no update:
153         float sum = oldPrefValue;
154         if (newPrefs != null) {
155           float newPrefValue = newPrefs.get(itemID);
156           if (!Float.isNaN(newPrefValue)) {
157             sum += newPrefValue;
158           }
159         }
160 
161         boolean remove = false;
162         if (removedItemIDs != null && removedItemIDs.contains(itemID)) {
163           remove = true;
164         } else if (FastMath.abs(sum) <= zeroThreshold) {
165           remove = true;
166         }
167 
168         if (!remove) {
169           emitter.emit(Pair.of(userID, new NumericIDValue(itemID, sum)));
170         }
171       }
172     }
173 
174     // Now output new data, that didn't exist in old prefs
175     if (newPrefs != null && !newPrefs.isEmpty()) {
176       for (LongFloatMap.MapEntry entry : newPrefs.entrySet()) {
177         long itemID = entry.getKey();
178         if (oldPrefs == null || !oldPrefs.containsKey(itemID)) {
179           // It wasn't already written. If it exists in newPrefs, it's also not removed
180           float newPrefValue = entry.getValue();
181           if (FastMath.abs(newPrefValue) > zeroThreshold) {
182             emitter.emit(Pair.of(userID, new NumericIDValue(itemID, newPrefValue)));
183           }
184         }
185       }
186     }
187   }
188 }
