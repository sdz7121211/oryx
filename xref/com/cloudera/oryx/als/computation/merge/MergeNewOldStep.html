18  import java.io.IOException;
19

21  import com.cloudera.oryx.als.computation.types.ALSTypes;
22  import com.google.common.base.Preconditions;
23  import org.apache.crunch.GroupingOptions;
24  import org.apache.crunch.PCollection;
25  import org.apache.crunch.PTable;
26  import org.apache.crunch.Pair;
27  import org.apache.crunch.impl.mr.MRPipeline;
28  import org.apache.crunch.lib.join.JoinUtils;
29  import org.apache.crunch.types.PTableType;
30  import org.apache.crunch.types.avro.Avros;
31  
32  import com.cloudera.oryx.als.computation.ALSJobStep;
33  import com.cloudera.oryx.computation.common.JobStepConfig;

35  import com.cloudera.oryx.common.servcomp.Store;

37  /**
38   * @author Sean Owen
39   */
40  public final class MergeNewOldStep extends ALSJobStep {
41  
42    private static final PTableType<Pair<Long, Integer>, NumericIDValue> JOIN_TYPE = Avros.tableOf(
43        Avros.pairs(ALSTypes.LONGS, ALSTypes.INTS),
44        ALSTypes.IDVALUE);
45  
46    @Override
47    protected MRPipeline createPipeline() throws IOException {
48  
49      JobStepConfig jobConfig = getConfig();
50  
51      String instanceDir = jobConfig.getInstanceDir();
52      int generationID = jobConfig.getGenerationID();
53      int lastGenerationID = jobConfig.getLastGenerationID();
54  
55      String outputKey = Namespaces.getInstanceGenerationPrefix(instanceDir, generationID) + "input/";
56      if (!validOutputPath(outputKey)) {
57        return null;
58      }
59  
60      MRPipeline p = createBasicPipeline(MergeNewOldValuesFn.class);
61  
62      String trainKey = Namespaces.getInstanceGenerationPrefix(instanceDir, generationID) + "train/";
63  
64      PCollection<Pair<Long, NumericIDValue>> parsed = p.read(textInput(trainKey))
65          .parallelDo("trainParse", new DelimitedInputParseFn(),
66              Avros.pairs(ALSTypes.LONGS, ALSTypes.IDVALUE));
67  
68      PTable<Pair<Long, Integer>, NumericIDValue> train = parsed.parallelDo("train", new InboundJoinFn(), JOIN_TYPE);
69  
70      if (lastGenerationID >= 0) {
71        String inputPrefix = Namespaces.getInstanceGenerationPrefix(instanceDir, lastGenerationID) + "input/";
72        Store store = Store.get();
73        Preconditions.checkState(store.exists(inputPrefix, false), "Input path does not exist: %s", inputPrefix);
74        PTable<Pair<Long, Integer>, NumericIDValue> joinInputBefore = p.read(input(inputPrefix, ALSTypes.VALUE_MATRIX))
75            .parallelDo("lastGenerationInput", new JoinBeforeMapFn(), JOIN_TYPE);
76        train = train.union(joinInputBefore);
77  
78        String testPrefix = Namespaces.getInstanceGenerationPrefix(instanceDir, lastGenerationID) + "test/";
79        if (store.exists(testPrefix, false)) {
80          // Held out test data last time; incorporate now
81          PTable<Pair<Long, Integer>, NumericIDValue> joinTestSetBefore =
82              p.read(textInput(testPrefix))
83                  .parallelDo("testParse", new DelimitedInputParseFn(), Avros.pairs(ALSTypes.LONGS, ALSTypes.IDVALUE))
84                  .parallelDo("lastGenerationTest", new JoinBeforeMapFn(), JOIN_TYPE);
85          train = train.union(joinTestSetBefore);
86        }
87      }
88  
89      GroupingOptions groupingOptions = GroupingOptions.builder()
90          .partitionerClass(JoinUtils.getPartitionerClass(train.getTypeFamily()))
91          .numReducers(getNumReducers())
92          .build();
93  
94      train
95          .groupByKey(groupingOptions)
96          .parallelDo(new MergeNewOldValuesFn(), ALSTypes.VALUE_MATRIX)
97          .write(output(outputKey));
98      return p;
99    }
100 
101   public static void main(String[] args) throws Exception {
102     run(new MergeNewOldStep(), args);
103   }
104 
105 }
