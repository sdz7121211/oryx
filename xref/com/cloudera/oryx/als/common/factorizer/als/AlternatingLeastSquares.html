52  /**
53   * <p>Implements the Alternating Least Squares algorithm described in
54   * <a href="http://www2.research.att.com/~yifanhu/PUB/cf.pdf">"Collaborative Filtering for Implicit Feedback Datasets"</a>
55   * by Yifan Hu, Yehuda Koren, and Chris Volinsky.</p>
56   *
57   * <p>This implementation varies in some small details; it does not use the same mechanism for explaining ratings
58   * for example and seeds the initial Y differently.</p>
59   *
60   * <p>Note that in this implementation, matrices are sparse and are implemented with a {@link com.cloudera.oryx.common.collection.LongObjectMap} of
61   * {@link com.cloudera.oryx.common.collection.LongFloatMap} so as to be able to use {@code long} keys. In many cases, a tall, skinny matrix is
62   * needed (sparse rows, dense columns). This is represented with {@link com.cloudera.oryx.common.collection.LongObjectMap} of {@code float[]}.</p>
63   *
64   * <p>This implementation implements essentially this function, expressed in Octave/Matlab:</p>
65   *
66   * {@code
67   * function [X, Y] = als(R, f, iterations)
68   *
69   *   alpha = 1;
70   *   lambda = 0.1 * alpha;
71   *   [m,n] = size(R);
72   *
73   *   P = R > 0;
74   *   Pnonzero = R != 0;
75   *   C = 1 + alpha * abs(R);
76   *
77   *   X = zeros(m,f);
78   *   Y = abs(stdnormal_rnd(n,f));
79   *   for i = 1:n
80   *     Y(i,:) /= norm(Y(i,:));
81   *   endfor
82   *
83   *   If = eye(f);
84   *   Im = eye(m);
85   *   In = eye(n);
86   *
87   *   for iter = 1:iterations
88   *
89   *     YTY = Y' * Y;
90   *     for u = 1:m
91   *       Cu = diag(C(u,:));
92   *       count = sum(Pnonzero(u,:));
93   *       YTCupu = Y' * Cu * P(u,:)';
94   *       YTCuY = YTY + Y' * (Cu - In) * Y;
95   *       Wu = pinv(YTCuY + lambda * count * If);
96   *       X(u,:) = (Wu * YTCupu)';
97   *     endfor
98   *
99   *     XTX = X' * X;
100  *     for i = 1:n
101  *       Ci = diag(C(:,i));
102  *       count = sum(Pnonzero(:,i));
103  *       XTCipi = X' * Ci * P(:,i);
104  *       XTCiX = XTX + X' * (Ci - Im) * X;
105  *       Wi = pinv(XTCiX + lambda * count * If);
106  *       Y(i,:) = (Wi * XTCipi)';
107  *     endfor
108  *
109  *   endfor
110  *
111  * end}
112  *
113  * @author Sean Owen
114  */
115 public final class AlternatingLeastSquares implements MatrixFactorizer {

119   private static final int WORK_UNIT_SIZE = 100;

123   private static final int MAX_FAR_FROM_VECTORS = 100000;

125   private final LongObjectMap<LongFloatMap> RbyRow;
126   private final LongObjectMap<LongFloatMap> RbyColumn;

134   /**
135    * @param RbyRow the input R matrix, indexed by row
136    * @param RbyColumn the input R matrix, indexed by column
137    * @param features number of features, must be positive
138    * @param estimateErrorConvergenceThreshold when the average absolute difference in estimated user-item
139    *   scores falls below this threshold between iterations, iterations will stop
140    * @param maxIterations caps the number of iterations run. If non-positive, there is no cap.
141    */
142   public AlternatingLeastSquares(LongObjectMap<LongFloatMap> RbyRow,
143                                  LongObjectMap<LongFloatMap> RbyColumn,
144                                  int features,
145                                  double estimateErrorConvergenceThreshold,
146                                  int maxIterations) {
147     Preconditions.checkNotNull(RbyRow);
148     Preconditions.checkNotNull(RbyColumn);
149     Preconditions.checkArgument(features > 0, "features must be positive: %s", features);
150     Preconditions.checkArgument(estimateErrorConvergenceThreshold > 0.0 && estimateErrorConvergenceThreshold < 1.0,
151                                 "threshold must be in (0,1): %s", estimateErrorConvergenceThreshold);
152     this.RbyRow = RbyRow;
153     this.RbyColumn = RbyColumn;
154     this.features = features;
155     this.estimateErrorConvergenceThreshold = estimateErrorConvergenceThreshold;
156     this.maxIterations = maxIterations;
157   }

178   @Override
179   public Void call() throws ExecutionException, InterruptedException {
180 
181     X = new LongObjectMap<float[]>(RbyRow.size());
182 
183     boolean randomY = previousY == null || previousY.isEmpty();
184     Y = constructInitialY(previousY);
185 
186     // This will be used to compute rows/columns in parallel during iteration
187 
188     int numThreads = ExecutorUtils.getParallelism();
189     ExecutorService executor =
190         Executors.newFixedThreadPool(numThreads,
191                                      new ThreadFactoryBuilder().setNameFormat("ALS-%d").setDaemon(true).build());
192 
193     log.info("Iterating using {} threads", numThreads);
194 
195     RandomGenerator random = RandomManager.getRandom();
196     long[] testUserIDs = RandomUtils.chooseAboutNFromStream(NUM_USER_ITEMS_TO_TEST_CONVERGENCE, 
197                                                             RbyRow.keySetIterator(), 
198                                                             RbyRow.size(), 
199                                                             random);
200     long[] testItemIDs = RandomUtils.chooseAboutNFromStream(NUM_USER_ITEMS_TO_TEST_CONVERGENCE, 
201                                                             RbyColumn.keySetIterator(), 
202                                                             RbyColumn.size(), 
203                                                             random);
204     double[][] estimates = new double[testUserIDs.length][testItemIDs.length];
205     if (!X.isEmpty()) {
206       for (int i = 0; i < testUserIDs.length; i++) {
207         for (int j = 0; j < testItemIDs.length; j++) {
208           estimates[i][j] = SimpleVectorMath.dot(X.get(testUserIDs[i]), Y.get(testItemIDs[j])); 
209         }
210       }
211     }
212     // Otherwise X is empty because it's the first ever iteration. Estimates can be left at initial 0 value
213 
214     try {
215       int iterationNumber = 0;
216       while (true) {
217         iterateXFromY(executor);
218         iterateYFromX(executor);
219         DoubleWeightedMean averageAbsoluteEstimateDiff = new DoubleWeightedMean();
220         for (int i = 0; i < testUserIDs.length; i++) {
221           for (int j = 0; j < testItemIDs.length; j++) {
222             double newValue = SimpleVectorMath.dot(X.get(testUserIDs[i]), Y.get(testItemIDs[j]));            
223             double oldValue = estimates[i][j];
224             estimates[i][j] = newValue;
225             // Weight, simplistically, by newValue to emphasize effect of good recommendations.
226             // But that only makes sense where newValue > 0
227             if (newValue > 0.0f) {
228               averageAbsoluteEstimateDiff.increment(FastMath.abs(newValue - oldValue), newValue);
229             }
230           }
231         }
232       
233         iterationNumber++;
234         log.info("Finished iteration {}", iterationNumber);
235         if (maxIterations > 0 && iterationNumber >= maxIterations) {
236           log.info("Reached iteration limit");
237           break;
238         }
239         log.info("Avg absolute difference in estimate vs prior iteration: {}", averageAbsoluteEstimateDiff);
240         double convergenceValue = averageAbsoluteEstimateDiff.getResult();
241         if (!Doubles.isFinite(convergenceValue)) {
242           log.warn("Invalid convergence value, aborting iteration! {}", convergenceValue);
243           break;
244         }
245         // Don't converge after 1 iteration if starting from a random point
246         if (!(randomY && iterationNumber == 1) && convergenceValue < estimateErrorConvergenceThreshold) {
247           log.info("Converged");          
248           break;
249         }
250       }
251     } finally {
252       ExecutorUtils.shutdownNowAndAwait(executor);
253     }
254     return null;
255   }

257   private LongObjectMap<float[]> constructInitialY(LongObjectMap<float[]> previousY) {
258 
259     RandomGenerator random = RandomManager.getRandom();
260     
261     LongObjectMap<float[]> randomY;
262     if (previousY == null || previousY.isEmpty()) {
263       // Common case: have to start from scratch
264       log.info("Starting from new, random Y matrix");      
265       randomY = new LongObjectMap<float[]>(RbyColumn.size());
266       
267     } else {
268       
269       int oldFeatureCount = previousY.entrySet().iterator().next().getValue().length;
270       if (oldFeatureCount > features) {
271         // Fewer features, use some dimensions from prior larger number of features as-is
272         log.info("Feature count has decreased to {}, projecting down previous generation's Y matrix", features);                
273         randomY = new LongObjectMap<float[]>(previousY.size());
274         for (LongObjectMap.MapEntry<float[]> entry : previousY.entrySet()) {
275           float[] oldLargerVector = entry.getValue();
276           float[] newSmallerVector = new float[features];
277           System.arraycopy(oldLargerVector, 0, newSmallerVector, 0, newSmallerVector.length);
278           SimpleVectorMath.normalize(newSmallerVector);
279           randomY.put(entry.getKey(), newSmallerVector);
280         }
281         
282       } else if (oldFeatureCount < features) {
283         log.info("Feature count has increased to {}, using previous generation's Y matrix as subspace", features);        
284         randomY = new LongObjectMap<float[]>(previousY.size());
285         for (LongObjectMap.MapEntry<float[]> entry : previousY.entrySet()) {
286           float[] oldSmallerVector = entry.getValue();
287           float[] newLargerVector = new float[features];
288           System.arraycopy(oldSmallerVector, 0, newLargerVector, 0, oldSmallerVector.length);
289           // Fill in new dimensions with random values
290           for (int i = oldSmallerVector.length; i < newLargerVector.length; i++) {
291             newLargerVector[i] = FastMath.abs((float) random.nextGaussian());
292           }
293           SimpleVectorMath.normalize(newLargerVector);
294           randomY.put(entry.getKey(), newLargerVector);
295         }
296         
297       } else {
298         // Common case: previous generation is same number of features
299         log.info("Starting from previous generation's Y matrix");        
300         randomY = previousY;
301       }
302     }
303     
304     List<float[]> recentVectors = Lists.newArrayList();
305     for (LongObjectMap.MapEntry<float[]> entry : randomY.entrySet()) {
306       if (recentVectors.size() >= MAX_FAR_FROM_VECTORS) {
307         break;
308       }
309       recentVectors.add(entry.getValue());
310     }
311     LongPrimitiveIterator it = RbyColumn.keySetIterator();
312     long count = 0;
313     while (it.hasNext()) {
314       long id = it.nextLong();
315       if (!randomY.containsKey(id)) {
316         float[] vector = RandomUtils.randomUnitVectorFarFrom(features, recentVectors, random);
317         randomY.put(id, vector);
318         if (recentVectors.size() < MAX_FAR_FROM_VECTORS) {
319           recentVectors.add(vector);
320         }
321       }
322       if (++count % LOG_INTERVAL == 0) {
323         log.info("Computed {} initial Y rows", count);
324       }
325     }
326     log.info("Constructed initial Y");
327     return randomY;
328   }

330   /**
331    * Runs one iteration to compute X from Y.
332    */
333   private void iterateXFromY(ExecutorService executor) throws ExecutionException, InterruptedException {
334 
335     RealMatrix YTY = MatrixUtils.transposeTimesSelf(Y);
336     Collection<Future<?>> futures = Lists.newArrayList();
337     addWorkers(RbyRow, Y, YTY, X, executor, futures);
338 
339     int count = 0;
340     long total = 0;
341     for (Future<?> f : futures) {
342       f.get();
343       count += WORK_UNIT_SIZE;
344       if (count >= LOG_INTERVAL) {
345         total += count;
346         JVMEnvironment env = new JVMEnvironment();
347         log.info("{} X/tag rows computed ({}MB heap)", total, env.getUsedMemoryMB());
348         if (env.getPercentUsedMemory() > 95) {
349           log.warn("Memory is low. Increase heap size with -Xmx, decrease new generation size with larger " +
350                    "-XX:NewRatio value, and/or use -XX:+UseCompressedOops");
351         }
352         count = 0;
353       }
354     }
355   }

357   /**
358    * Runs one iteration to compute Y from X.
359    */
360   private void iterateYFromX(ExecutorService executor) throws ExecutionException, InterruptedException {
361 
362     RealMatrix XTX = MatrixUtils.transposeTimesSelf(X);
363     Collection<Future<?>> futures = Lists.newArrayList();
364     addWorkers(RbyColumn, X, XTX, Y, executor, futures);
365 
366     int count = 0;
367     long total = 0;
368     for (Future<?> f : futures) {
369       f.get();
370       count += WORK_UNIT_SIZE;
371       if (count >= LOG_INTERVAL) {
372         total += count;
373         JVMEnvironment env = new JVMEnvironment();
374         log.info("{} Y/tag rows computed ({}MB heap)", total, env.getUsedMemoryMB());
375         if (env.getPercentUsedMemory() > 95) {
376           log.warn("Memory is low. Increase heap size with -Xmx, decrease new generation size with larger " +
377                    "-XX:NewRatio value, and/or use -XX:+UseCompressedOops");
378         }
379         count = 0;
380       }
381     }
382   }

384   private void addWorkers(LongObjectMap<LongFloatMap> R,
385                           LongObjectMap<float[]> M,
386                           RealMatrix MTM, 
387                           LongObjectMap<float[]> MTags,
388                           ExecutorService executor,                          
389                           Collection<Future<?>> futures) {
390     if (R != null) {
391       List<Pair<Long, LongFloatMap>> workUnit = Lists.newArrayListWithCapacity(WORK_UNIT_SIZE);
392       for (LongObjectMap.MapEntry<LongFloatMap> entry : R.entrySet()) {
393         workUnit.add(new Pair<Long,LongFloatMap>(entry.getKey(), entry.getValue()));
394         if (workUnit.size() == WORK_UNIT_SIZE) {
395           futures.add(executor.submit(new Worker(features, M, MTM, MTags, workUnit)));
396           workUnit = Lists.newArrayListWithCapacity(WORK_UNIT_SIZE);
397         }
398       }
399       if (!workUnit.isEmpty()) {
400         futures.add(executor.submit(new Worker(features, M, MTM, MTags, workUnit)));
401       }
402     }
403   }

405   private static final class Worker implements Callable<Object> {
406 
407     private final int features;
408     private final LongObjectMap<float[]> Y;
409     private final RealMatrix YTY;
410     private final LongObjectMap<float[]> X;
411     private final Iterable<Pair<Long, LongFloatMap>> workUnit;
412 
413     private Worker(int features,
414                    LongObjectMap<float[]> Y,
415                    RealMatrix YTY,
416                    LongObjectMap<float[]> X,
417                    Iterable<Pair<Long, LongFloatMap>> workUnit) {
418       this.features = features;
419       this.Y = Y;
420       this.YTY = YTY;
421       this.X = X;
422       this.workUnit = workUnit;
423     }
424 
425     @Override
426     public Void call() {
427 
428       Config config = ConfigUtils.getDefaultConfig();
429       double alpha = config.getDouble("model.alpha");
430       double lambda = config.getDouble("model.lambda") * alpha;
431       // This will cause the ALS algorithm to reconstruction the input matrix R, rather than the
432       // matrix P = R > 0 . Don't use this unless you understand it!
433       boolean reconstructRMatrix = config.getBoolean("model.reconstruct-r-matrix");
434       // Causes the loss function to exclude entries for any input pairs that do not appear in the
435       // input and are implicitly 0
436       // Likewise, don't touch this for now unless you know what it does.
437       boolean lossIgnoresUnspecified = config.getBoolean("model.loss-ignores-unspecified");
438 
439       int features = this.features;
440       // Each worker has a batch of rows to compute:
441       for (Pair<Long,LongFloatMap> work : workUnit) {
442 
443         // Row (column) in original R matrix containing total association value. For simplicity we will
444         // talk about users and rows only in the comments and variables. It's symmetric for columns / items.
445         // This is Ru:
446         LongFloatMap ru = work.getSecond();
447 
448         // Start computing Wu = (YT*Cu*Y + lambda*I) = (YT*Y + YT*(Cu-I)*Y + lambda*I),
449         // by first starting with a copy of YT * Y. Or, a variant on YT * Y, if LOSS_IGNORES_UNSPECIFIED is set
450         RealMatrix Wu =
451             lossIgnoresUnspecified ?
452             partialTransposeTimesSelf(Y, YTY.getRowDimension(), ru.keySetIterator()) : 
453             YTY.copy();
454 
455         double[][] WuData = MatrixUtils.accessMatrixDataDirectly(Wu);
456         double[] YTCupu = new double[features];
457 
458         for (LongFloatMap.MapEntry entry : ru.entrySet()) {
459 
460           double xu = entry.getValue();
461 
462           float[] vector = Y.get(entry.getKey());
463           if (vector == null) {
464             log.warn("No vector for {}. This should not happen. Continuing...", entry.getKey());
465             continue;
466           }
467 
468           // Wu and YTCupu
469           if (reconstructRMatrix) {
470             for (int row = 0; row < features; row++) {
471               YTCupu[row] += xu * vector[row];
472             }
473           } else {
474             double cu = 1.0 + alpha * FastMath.abs(xu);            
475             for (int row = 0; row < features; row++) {
476               float vectorAtRow = vector[row];
477               double rowValue = vectorAtRow * (cu - 1.0);
478               double[] WuDataRow = WuData[row];              
479               for (int col = 0; col < features; col++) {
480                 WuDataRow[col] += rowValue * vector[col];
481                 //Wu.addToEntry(row, col, rowValue * vector[col]);
482               }
483               if (xu > 0.0) {
484                 YTCupu[row] += vectorAtRow * cu;
485               }
486             }
487           }
488 
489         }
490 
491         double lambdaTimesCount = lambda * ru.size();
492         for (int x = 0; x < features; x++) {
493           WuData[x][x] += lambdaTimesCount;          
494           //Wu.addToEntry(x, x, lambdaTimesCount);
495         }
496 
497         float[] xu = MatrixUtils.getSolver(Wu).solveDToF(YTCupu);
498 
499         // Store result:
500         synchronized (X) {
501           X.put(work.getFirst(), xu);
502         }
503 
504         // Process is identical for computing Y from X. Swap X in for Y, Y for X, i for u, etc.
505       }
506       return null;
507     }
508 
509     /**
510      * Like {@link MatrixUtils#transposeTimesSelf(com.cloudera.oryx.common.collection.LongObjectMap)}, but instead of computing MT * M,
511      * it computes MT * C * M, where C is a diagonal matrix of 1s and 0s. This is like pretending some
512      * rows of M are 0.
513      * 
514      * @see MatrixUtils#transposeTimesSelf(com.cloudera.oryx.common.collection.LongObjectMap)
515      */
516     private static RealMatrix partialTransposeTimesSelf(LongObjectMap<float[]> M,
517                                                         int dimension, 
518                                                         LongPrimitiveIterator keys) {
519       RealMatrix result = new Array2DRowRealMatrix(dimension, dimension);
520       while (keys.hasNext()) {
521         long key = keys.next();
522         float[] vector = M.get(key);
523         for (int row = 0; row < dimension; row++) {
524           float rowValue = vector[row];
525           for (int col = 0; col < dimension; col++) {
526             result.addToEntry(row, col, rowValue * vector[col]);
527           }
528         }
529       }
530       return result;
531     }
