16  package com.cloudera.oryx.common.iterator;
17  
18  import java.io.File;
19  import java.io.IOException;
20  import java.io.Reader;
21  import java.util.Iterator;
22  
23  import com.cloudera.oryx.common.io.IOUtils;
24  
25  /**
26   * Iterable representing the lines of a text file. It can produce an {@link Iterator} over those lines. This
27   * assumes the text file's lines are delimited in a manner consistent with how {@link java.io.BufferedReader}
28   * defines lines.
29   * 
30   * This class will uncompress files that end in .zip or .gz accordingly, too.
31   * 
32   * @author Sean Owen
33   */
34  public final class FileLineIterable implements Iterable<String> {
35  
36    private final Reader reader;
37  
38    /** 
39     * Creates a {@code FileLineIterable} over a given {@link File}, assuming a UTF-8 encoding.
40     */
41    public FileLineIterable(File file) throws IOException {
42      this.reader = IOUtils.openReaderMaybeDecompressing(file);
43    }
44  
45    /**
46     * Creates a {@code FileLineIterable} over a given stream of characters from a {@link Reader},
47     * assuming a UTF-8 encoding.
48     */
49    public FileLineIterable(Reader reader) {
50      this.reader = reader;
51    }
52    
53    @Override
54    public Iterator<String> iterator() {
55      return new FileLineIterator(reader);
56    }
57    
58  }
