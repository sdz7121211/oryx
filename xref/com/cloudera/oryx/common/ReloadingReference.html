16  package com.cloudera.oryx.common;
17  
18  import java.util.concurrent.Callable;
19  import java.util.concurrent.TimeUnit;
20  import java.util.concurrent.locks.Lock;
21  import java.util.concurrent.locks.ReentrantLock;
22  
23  import com.google.common.base.Preconditions;
24  import com.google.common.base.Supplier;
25  import org.slf4j.Logger;
26  import org.slf4j.LoggerFactory;
27  
28  /**
29   * Encapsulates a reference to something that is created the first time it is needed. Instead
30   * of providing an initial value, a {@link Callable} is given that can create the thing is provided.
31   * This can also periodically re-create or reload the value.
32   *
33   * @author Sean Owen
34   * @param <V> type of referent encapsulated and reloaded by this class
35   */
36  public final class ReloadingReference<V> implements Supplier<V> {
37  
38    private static final Logger log = LoggerFactory.getLogger(ReloadingReference.class);
39  
40    private static final long NO_RELOAD = -1L;
41  
42    private V value;
43    private final Callable<V> retriever;
44    private long lastRetrieval;
45    private final long originalDurationMS;
46    private long currentDurationMS;
47    private final Lock lock;
48  
49    /**
50     * A {@code ReloadingReference} that never reloads the reference after it is loaded once.
51     *
52     * @param retriever object that can supply a new, current value of the reference
53     */
54    public ReloadingReference(Callable<V> retriever) {
55      this(retriever, NO_RELOAD, null);
56    }
57  
58    /**
59     * A {@code ReloadingReference} that reloads if the loaded value was loaded longer ago than the given
60     * duration.
61     *
62     * @param retriever object that can supply a new, current value of the reference
63     * @param duration time after which a new value will be loaded, or {@link #NO_RELOAD} to not reload
64     *  after the initial value is loaded
65     * @param timeUnit units of time for duration
66     */
67    public ReloadingReference(Callable<V> retriever, long duration, TimeUnit timeUnit) {
68      Preconditions.checkNotNull(retriever);
69      this.retriever = retriever;
70      if (duration == NO_RELOAD) {
71        originalDurationMS = NO_RELOAD;
72      } else {
73        Preconditions.checkArgument(duration > 0, "Duration must be positive: %s", duration);
74        Preconditions.checkNotNull(timeUnit);
75        originalDurationMS = TimeUnit.MILLISECONDS.convert(duration, timeUnit);
76      }
77      currentDurationMS = originalDurationMS;
78      lock = new ReentrantLock();
79    }
80  
81    /**
82     * Like {@link #get()}, but when a value already exists, will only wait for the given amount of time before
83     * just proceeding to return the existing value instead of waiting for a new value to load.
84     */
85    public V get(long timeout, TimeUnit timeUnit) {
86      try {
87        if (lock.tryLock(timeout, timeUnit)) {
88          try {
89            doGet();
90          } finally {
91            lock.unlock();
92          }
93        }
94        // timed out
95      } catch (InterruptedException ie) {
96        // interrupted
97      }
98      V theValue = value;
99      return theValue == null ? get() : theValue;
100   }
101 
102   /**
103    * @return object that is returned by the provided {@link Callable}. If not yet created, it will block and
104    *  wait for creation. If already created, it will return the existing value. The value will be re-created
105    *  periodically, if the object has been configured to, and when this is needed, this method will again
106    *  block while the value is re-created.
107    */
108   @Override
109   public V get() {
110     lock.lock();
111     try {
112       doGet();
113     } finally {
114       lock.unlock();
115     }
116     return value;
117   }
118 
119   private void doGet() {
120     boolean reloading = originalDurationMS > 0L;
121     long now = reloading ? System.currentTimeMillis() : 0L;
122     if (value == null || (reloading && now > lastRetrieval + currentDurationMS)) {
123       try {
124         value = retriever.call();
125         Preconditions.checkState(value != null);
126       } catch (Exception e) {
127         // Kind of arbitrary exponential backoff -- 2x after each error up to 16x
128         // If too many errors, or error on first retrieval, die
129         if (currentDurationMS >= originalDurationMS * 16 || value == null) {
130           // Too many errors, start failing
131           throw new IllegalStateException(e);          
132         }
133         // else log and quietly back off
134         log.warn("Retrieval failed; using previous cached value", e);
135         currentDurationMS *= 2;
136       }
137       lastRetrieval = System.currentTimeMillis();      
138       // Reset backoff
139       currentDurationMS = originalDurationMS;      
140     }
141   }
142 
143   /**
144    * @return object that is returned by the provided {@link Callable}, if it has already been created
145    *  previously, or {@code null} otherwise
146    */
147   public V maybeGet() {
148     return value;
149   }
150 
151   /**
152    * Clears the reference, requiring a load on next access.
153    */
154   public void clear() {
155     lock.lock();
156     try {
157       value = null;
158     } finally {
159       lock.unlock();
160     }
161   }
162 
163 }
