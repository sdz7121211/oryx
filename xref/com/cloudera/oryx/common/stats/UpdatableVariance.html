16  package com.cloudera.oryx.common.stats;
17  
18  import com.google.common.base.Preconditions;
19  
20  /**
21   * Based on Welford's algorithm (see http://www.johndcook.com/standard_deviation.html) for numeric stability.
22   * We'd use Commons Math3 {@code Variance}, except that it doesn't allow decrementing.
23   *
24   * @author Sean Owen
25   */
26  public final class UpdatableVariance {
27  
28    private long n;
29    private double mk;
30    private double sk;
31  
32    public UpdatableVariance() {
33      mk = Double.NaN;
34      sk = Double.NaN;
35    }
36  
37    /**
38     * @param xk new datum to add to the variance computation
39     */
40    public void increment(double xk) {
41      if (++n == 1) {
42        mk = xk;
43        sk = 0.0;
44      } else {
45        double diff = xk - mk;
46        mk += diff / n;
47        sk += diff * (xk - mk);
48      }
49    }
50  
51    /**
52     * @param xk new datum to remove from the variance computation
53     */
54    public void decrement(double xk) {
55      Preconditions.checkState(n > 0);
56      if (--n == 0) {
57        mk = Double.NaN;
58        sk = Double.NaN;
59      } else {
60        double mkPlus1 = mk;
61        mk = ((n+1) * mkPlus1 - xk) / n;
62        sk -= (xk - mk) * (xk - mkPlus1);
63      }
64    }
65  
66    /**
67     * @return number of data points in the variance computation
68     */
69    public long getN() {
70      return n;
71    }
72  
73    /**
74     * @return variance of data points
75     */
76    public double getResult() {
77      return n <= 1 ? Double.NaN : sk / (n - 1);
78    }
79  
80    @Override
81    public String toString() {
82      return Double.toString(getResult());
83    }
84  
85  }
