39    /** The largest prime less than 2<sup>31</sup>-1 that is the smaller of a twin prime pair. */
40    public static final int MAX_INT_SMALLER_TWIN_PRIME = 2147482949;
41  
42    private static final HashFunction HASH = Hashing.md5();
43  
44    private RandomUtils() {
45    }
46  
47    /**
48     * @param dimensions dimensionality of resulting vector
49     * @param random random number generator to use
50     * @return a vector of length 1 over the given number of dimensions, whose direction is chosen uniformly
51     *   at random (that is: a point chosen uniformly at random on the unit hypersphere)
52     */
53    public static float[] randomUnitVector(int dimensions, RandomGenerator random) {
54      float[] vector = new float[dimensions];
55      doRandomUnitVector(vector, random);
56      return vector;
57    }
58    
59    private static void doRandomUnitVector(float[] vector, RandomGenerator random) {
60      int dimensions = vector.length;
61      double total = 0.0;
62      for (int i = 0; i < dimensions; i++) {
63        double d = random.nextGaussian();
64        vector[i] = (float) d;
65        total += d * d;
66      }
67      float normalization = (float) FastMath.sqrt(total);
68      for (int i = 0; i < dimensions; i++) {
69        vector[i] /= normalization;
70      }
71    }
72  
73    /**
74     * @param dimensions dimensionality of resulting vector
75     * @param farFrom vectors that the chosen vector should be "far from" -- not in the same direction as
76     * @param random random number generator to use
77     * @return a vector of length 1 over the given number of dimensions, whose direction is chosen uniformly
78     *   at random (that is: a point chosen uniformly at random on the unit hypersphere), but preferring
79     *   those not in the same direction as a set of existing vectors
80     */
81    public static float[] randomUnitVectorFarFrom(int dimensions,
82                                                  List<float[]> farFrom,
83                                                  RandomGenerator random) {
84      int size = farFrom.size();
85      int numSamples = FastMath.min(100, size);
86      float[] vector = new float[dimensions];
87      boolean accepted = false;
88      while (!accepted) {
89        doRandomUnitVector(vector, random);
90        double smallestDistSquared = Double.POSITIVE_INFINITY;
91        for (int sample = 0; sample < numSamples; sample++) {
92          float[] other = farFrom.get(size == numSamples ? sample : random.nextInt(size));
93          // dot is the cosine here since both are unit vectors
94          double distSquared = 2.0 - 2.0 * SimpleVectorMath.dot(vector, other);
95          if (Doubles.isFinite(distSquared) && distSquared < smallestDistSquared) {
96            smallestDistSquared = distSquared;
97          }
98        }
99        // Second condition covers 1-D case, where there are only 2 distinct unit vectors. If both have
100       // been generated, keep accepting either of them.
101       if (Doubles.isFinite(smallestDistSquared) && !(dimensions == 1 && smallestDistSquared == 0.0)) {
102         // Choose with probability proportional to squared distance, a la kmeans++ centroid selection
103         double acceptProbability = smallestDistSquared / 4.0; // dist squared is in [0,4]
104         accepted = random.nextDouble() < acceptProbability;
105       } else {
106         // kind of a default
107         accepted = true;
108       }
109     }
110     return vector;
111   }

113   /**

115    */
116   public static long hash(CharSequence value) {
117     return HASH.hashString(value).asLong();
118   }

120   /**
121    * Finds next-largest "twin primes": numbers p and p+2 such that both are prime. Finds the smallest such p
122    * such that the smaller twin, p, is greater than or equal to n. Returns p+2, the larger of the two twins.
123    */
124   public static int nextTwinPrime(int n) {
125     if (n > MAX_INT_SMALLER_TWIN_PRIME) {
126       throw new IllegalArgumentException();
127     }
128     if (n <= 3) {
129       return 5;
130     }
131     int next = Primes.nextPrime(n);
132     while (!Primes.isPrime(next + 2)) {
133       next = Primes.nextPrime(next + 4);
134     }
135     return next + 2;
136   }

138   /**
139    * @param n approximate number of items to choose
140    * @param stream stream to choose from randomly
141    * @param streamSize (approximate) stream size
142    * @param random random number generator
143    * @return up to n elements chosen uninformly at random from the stream
144    */
145   public static long[] chooseAboutNFromStream(int n, 
146                                               LongPrimitiveIterator stream,
147                                               int streamSize, 
148                                               RandomGenerator random) {
149     LongPrimitiveIterator it;
150     if (n < streamSize) {
151       it = new SamplingLongPrimitiveIterator(random, stream, (double) n / streamSize);      
152     } else {
153       it = stream;
154     }
155     LongSet chosen = new LongSet(n);
156     while (it.hasNext()) {
157       chosen.add(it.nextLong());
158     }
159     return chosen.toArray();
160   }
