88    @Override
89    public final Collection<StepState> getStepStates() throws IOException {
90      String name = getCustomJobName();
91      StepStatus status = determineStatus();
92      StepState state = new StepState(startTime, endTime, name, status);
93      if (status == StepStatus.COMPLETED) {
94        // This completed, -- perhaps without ever running, which means Hadoop doesn't report state because
95        // it was skipped. It's complete.
96        state.setMapProgress(1.0f);
97        state.setReduceProgress(1.0f);
98      } else {
99        float[] progresses = determineProgresses();
100       if (progresses != null) {
101         // 0 is setup progress; we don't care
102         state.setMapProgress(progresses[1]);
103         state.setReduceProgress(progresses[2]);
104       }
105     }
106     return Collections.singletonList(state);
107   }
108 
109   private StepStatus determineStatus() throws IOException {
110     JobContext job = getJob();
111     if (job == null) {
112       return StepStatus.COMPLETED;
113     }
114     JobClient client = new JobClient(new JobConf(getConf()));
115     try {
116       JobID jobID = job.getJobID();
117       if (jobID == null) {
118         return StepStatus.PENDING;
119       }
120       RunningJob runningJob =
121           client.getJob(new org.apache.hadoop.mapred.JobID(jobID.getJtIdentifier(), jobID.getId()));
122       if (runningJob == null) {
123         return StepStatus.PENDING;
124       }
125       int state = runningJob.getJobState();
126       // Can't use switch since these are no longer compile-time constants in Hadoop 2.x
127       if (state == JobStatus.PREP) {
128         return StepStatus.PENDING;
129       }
130       if (state == JobStatus.RUNNING) {
131         return StepStatus.RUNNING;
132       }
133       if (state == JobStatus.FAILED) {
134         return StepStatus.FAILED;
135       }
136       if (state == JobStatus.KILLED) {
137         return StepStatus.CANCELLED;
138       }
139       if (state == JobStatus.SUCCEEDED) {
140         return StepStatus.COMPLETED;
141       }
142       throw new IllegalArgumentException("Unknown Hadoop job state " + state);
143     } finally {
144       client.close();
145     }
146   }

148   private JobContext getJob() {
149     return exec == null ? null : exec.getJobs().get(0).getJob();
150   }

152   /**
153    * @return three progress values, in [0,1], as a {@code float[]}, representing setup, mapper and reducer progress
154    */
155   private float[] determineProgresses() throws IOException {
156     if (exec == null) {
157       return null;
158     }
159     JobClient client = new JobClient(new JobConf(getConf()));
160     try {
161       JobID jobID = getJob().getJobID();
162       if (jobID == null) {
163         return null;
164       }
165       RunningJob runningJob =
166           client.getJob(new org.apache.hadoop.mapred.JobID(jobID.getJtIdentifier(), jobID.getId()));
167       if (runningJob == null) {
168         return null;
169       }
170 
171       return new float[] { runningJob.setupProgress(), runningJob.mapProgress(), runningJob.reduceProgress() };
172     } finally {
173       client.close();
174     }
175   }

177   @Override
178   public final int run(String[] args) throws InterruptedException, IOException, JobException {
179 
180     MemoryHandler.setSensibleLogFormat();
181 
182     if (log.isInfoEnabled()) {
183       log.info("args are {}", Arrays.toString(args));
184     }
185 
186     config = parseConfig(args);
187 
188     String name = getCustomJobName();
189     log.info("Running {}", name);
190 
191     long start = System.currentTimeMillis();
192     startTime = new Date(start);
193     MRPipeline pipeline = createPipeline();
194     try {
195       if (pipeline == null) {
196         log.info("{} does not need to run", name);
197       } else {
198 
199         exec = pipeline.runAsync();
200         log.info("Waiting for {} to complete", name);
201         exec.waitUntilDone();
202 
203         PipelineExecution.Status exitStatus = exec.getStatus();
204         if (exitStatus == PipelineExecution.Status.KILLED || exitStatus == PipelineExecution.Status.FAILED) {
205           exec.kill();
206           throw new JobException(name + " failed in state " + exitStatus);
207         }
208 
209         log.info("Finished {}", name);
210         postRun();
211 
212       }
213     } catch (InterruptedException ie) {
214       log.warn("Interrupted {}", name);
215       exec.kill();
216       throw ie;
217     } finally {
218       long end = System.currentTimeMillis();
219       endTime = new Date(end);
220       log.info("Completed {} in {}s", this, (end - start) / 1000L);
221       if (pipeline != null) {
222         pipeline.done();
223       }
224     }
225 
226     return 0;
227   }

250   /**
251    * Creates a new {@link MRPipeline} instance that contains common configuration
252    * settings.
253    *
254    * @return a new {@link MRPipeline} instance, suitably configured
255    */
256   protected final MRPipeline createBasicPipeline(Class<?> jarClass) throws IOException {
257     Configuration conf = OryxConfiguration.get(getConf());
258 
259     conf.setBoolean("mapred.map.output.compress", true);
260     conf.setClass("mapred.map.output.compress.codec", SnappyCodec.class, CompressionCodec.class);
261 
262     conf.setBoolean("mapred.output.compress", true);
263     conf.set("mapred.output.compression.type", "BLOCK");
264     conf.setClass("mapred.output.compression.codec", SnappyCodec.class, CompressionCodec.class);
265     // Set old-style equivalents for Avro/Crunch's benefit
266     conf.set("avro.output.codec", "snappy");
267 
268     conf.setBoolean("mapreduce.map.speculative", true);
269     conf.setBoolean("mapreduce.reduce.speculative", true);
270     conf.setBoolean("mapreduce.tasktracker.outofband.heartbeat", true);
271     conf.setInt("mapreduce.job.jvm.numtasks", -1);
272 
273     //conf.setBoolean("crunch.disable.deep.copy", true);
274 
275     Config appConfig = ConfigUtils.getDefaultConfig();
276 
277     int mapMemoryMB = appConfig.getInt("computation-layer.mapper-memory-mb");
278     log.info("Mapper memory: {}", mapMemoryMB);
279     int mapHeapMB = (int) (mapMemoryMB / 1.3); // Matches Hadoop's default
280     log.info("Mappers have {}MB heap and can access {}MB RAM", mapHeapMB, mapMemoryMB);
281     if (conf.get("mapreduce.map.java.opts") != null) {
282       log.info("Overriding previous setting of {}, which was '{}'",
283                "mapreduce.map.java.opts",
284                conf.get("mapreduce.map.java.opts"));
285     }
286     conf.set("mapreduce.map.java.opts",
287              "-Xmx" + mapHeapMB + "m -XX:+UseCompressedOops -XX:+UseParallelGC -XX:+UseParallelOldGC");
288     log.info("Set {} to '{}'", "mapreduce.map.java.opts", conf.get("mapreduce.map.java.opts"));
289     // See comment below on CM
290     conf.setInt("mapreduce.map.java.opts.max.heap", mapHeapMB);
291 
292     int reduceMemoryMB = appConfig.getInt("computation-layer.reducer-memory-mb");
293     log.info("Reducer memory: {}", reduceMemoryMB);
294     if (isHighMemoryStep()) {
295       reduceMemoryMB *= appConfig.getInt("computation-layer.worker-high-memory-factor");
296       log.info("Increasing {} to {} for high-memory step", "mapreduce.reduce.memory.mb", reduceMemoryMB);
297     }
298     conf.setInt("mapreduce.reduce.memory.mb", reduceMemoryMB);
299 
300     int reduceHeapMB = (int) (reduceMemoryMB / 1.3); // Matches Hadoop's default
301     log.info("Reducers have {}MB heap and can access {}MB RAM", reduceHeapMB, reduceMemoryMB);
302     if (conf.get("mapreduce.reduce.java.opts") != null) {
303       log.info("Overriding previous setting of {}, which was '{}'",
304                "mapreduce.reduce.java.opts",
305                conf.get("mapreduce.reduce.java.opts"));
306     }
307     conf.set("mapreduce.reduce.java.opts",
308              "-Xmx" + reduceHeapMB + "m -XX:+UseCompressedOops -XX:+UseParallelGC -XX:+UseParallelOldGC");
309     log.info("Set {} to '{}'", "mapreduce.reduce.java.opts", conf.get("mapreduce.reduce.java.opts"));
310     // I see this in CM but not in Hadoop docs; probably won't hurt as it's supposed to result in
311     // -Xmx appended to opts above, which is at worst redundant
312     conf.setInt("mapreduce.reduce.java.opts.max.heap", reduceHeapMB);
313 
314     conf.setInt("yarn.scheduler.capacity.minimum-allocation-mb", 128);
315     conf.setInt("yarn.app.mapreduce.am.resource.mb", 384);
316 
317     // Pass total config state
318     conf.set(CONFIG_SERIALIZATION_KEY, ConfigUtils.getDefaultConfig().root().render());
319 
320     // Make sure to set any args to conf above this line!
321 
322     setConf(conf);
323 
324     Job job = Job.getInstance(conf);
325 
326     // Basic File IO settings
327     FileInputFormat.setMaxInputSplitSize(job, 1L << 30); // ~1073MB
328     FileInputFormat.setMinInputSplitSize(job, 1L << 27); // ~134MB
329     SequenceFileOutputFormat.setOutputCompressionType(job, SequenceFile.CompressionType.BLOCK);
330     FileOutputFormat.setCompressOutput(job, true);
331     FileOutputFormat.setOutputCompressorClass(job, SnappyCodec.class);
332 
333     log.info("Created pipeline configuration {}", job.getConfiguration());
334 
335     return new MRPipeline(jarClass, getCustomJobName(), job.getConfiguration());
336   }

338   protected final boolean validOutputPath(String outputPathKey) throws IOException {
339     Preconditions.checkArgument(outputPathKey != null && outputPathKey.endsWith("/"),
340         "%s should end with /", outputPathKey);
341 
342     String jobName = getCustomJobName();
343 
344     Store store = Store.get();
345     if (store.exists(outputPathKey, false)) {
346       log.info("Output path {} already exists", outputPathKey);
347       if (store.exists(outputPathKey + "_SUCCESS", true)) {
348         log.info("{} shows _SUCCESS present; skipping", jobName);
349         return false;
350       }
351       log.info("{} seems to have stopped during an earlier run; deleting output at {} and recomputing",
352           jobName, outputPathKey);
353       store.recursiveDelete(outputPathKey);
354     } else {
355       log.info("Output path is clear for writing: {}", outputPathKey);
356     }
357     return true;
358   }

403   protected final GroupingOptions groupingOptions(
404       Class<? extends Partitioner> partitionerClass,
405       Class<? extends RawComparator<?>> groupingComparator,
406       Class<? extends RawComparator<?>> sortComparator) {
407     GroupingOptions.Builder b = GroupingOptions.builder()
408         .partitionerClass(partitionerClass)
409         .numReducers(getNumReducers());
410 
411     if (groupingComparator != null) {
412       b.groupingComparatorClass(groupingComparator);
413     }
414     if (sortComparator != null) {
415       b.sortComparatorClass(sortComparator);
416     }
417     return b.build();
418   }

420   protected final int getNumReducers() {
421     String parallelismString = ConfigUtils.getDefaultConfig().getString("computation-layer.parallelism");
422     if ("auto".equals(parallelismString)) {
423       return 11; // Default to a prime. Helps avoid problems with funny distributions of IDs.
424     } else {
425       return Integer.parseInt(parallelismString);
426     }
427   }

429   public static void run(Tool step, String[] args) throws IOException, InterruptedException, JobException {
430     log.info("Running step {}", step.getClass().getSimpleName());
431     try {
432       int result = ToolRunner.run(OryxConfiguration.get(), step, args);
433       if (result != 0) {
434         throw new JobException(step + " hasd bad exit status: " + result);
435       }
436     } catch (IOException ioe) {
437       throw ioe;
438     } catch (InterruptedException ie) {
439       throw ie;
440     } catch (JobException je) {
441       throw je;
442     } catch (Exception e) {
443       throw new JobException(e);
444     }
445   }
