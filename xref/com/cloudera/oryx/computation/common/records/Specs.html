16  package com.cloudera.oryx.computation.common.records;
17  
18  import java.util.List;
19  
20  import com.google.common.collect.ImmutableList;
21  import com.google.common.collect.Lists;
22  
23  /**
24   * Utility functions for working with header files and {@code Spec} data.
25   */
26  public final class Specs {
27  
28    public static boolean isNumeric(Spec spec, String fieldId) {
29      FieldSpec fs = spec.getField(getFieldId(spec, fieldId));
30      return fs.spec().getDataType().isNumeric();
31    }
32    
33    public static Integer getFieldId(Spec spec, String value) {
34      List<Integer> fieldIds = getFieldIds(spec, ImmutableList.of(value));
35      if (fieldIds.isEmpty()) {
36        throw new IllegalArgumentException("Could not find field " + value + " in spec");
37      }
38      return fieldIds.get(0);
39    }
40  
41    public static List<Integer> getFieldIds(Spec spec, List<String> values) {
42      if (values.isEmpty()) {
43        return ImmutableList.of();
44      }
45      List<Integer> fieldIds = Lists.newArrayListWithExpectedSize(values.size());
46      if (spec == null || spec.getField(values.get(0)) == null) {
47        for (String value : values) {
48          try {
49            fieldIds.add(Integer.valueOf(value));
50          } catch (NumberFormatException e) {
51            throw new IllegalArgumentException("Did not recognize column ID: " + value);
52          }
53        }
54      } else {
55        for (String value : values) {
56          FieldSpec f = spec.getField(value);
57          if (f != null) {
58            fieldIds.add(f.position());
59          }
60        }
61      }
62      return fieldIds;
63    }
64    
65    private Specs() {
66    }
67  }
