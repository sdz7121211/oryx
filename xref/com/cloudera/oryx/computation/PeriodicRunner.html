57  /**
58   * <p>This class will periodically run one generation of the Computation Layer.
59   * It can run after a period of time has elapsed, or an amount of data has been written.</p>
60   *
61   * @author Sean Owen
62   */
63  public final class PeriodicRunner implements Runnable, Closeable {
64    // Implement Runnable, not Callable, for ScheduledExecutor
65  
66    private static final Logger log = LoggerFactory.getLogger(PeriodicRunner.class);
67  
68    private static final Map<String,List<Class<? extends GenerationRunner>>> RUNNERS;
69    static {
70      List<Class<? extends GenerationRunner>> alsRunners = Lists.newArrayListWithCapacity(2);
71      alsRunners.add(ALSLocalGenerationRunner.class);
72      alsRunners.add(ALSDistributedGenerationRunner.class);
73      List<Class<? extends GenerationRunner>> kmeansRunners = Lists.newArrayListWithCapacity(2);
74      kmeansRunners.add(KMeansLocalGenerationRunner.class);
75      kmeansRunners.add(KMeansDistributedGenerationRunner.class);
76      List<Class<? extends GenerationRunner>> rdfRunners = Lists.newArrayListWithCapacity(2);
77      rdfRunners.add(RDFLocalGenerationRunner.class);
78      rdfRunners.add(RDFDistributedGenerationRunner.class);
79      RUNNERS = ImmutableMap.of("als", alsRunners, "kmeans", kmeansRunners, "rdf", rdfRunners);
80    }
81  
82    private static final long CHECK_INTERVAL_MINS = 1;
83  
84    private final Config config;
85    private long lastUpdateTime;
86    private long nextScheduledRunTime;
87    private boolean forceRun;
88    private boolean running;
89    private final Collection<GenerationRunner> allGenerationRunners;
90    private final ScheduledExecutorService executor;
91    private Future<?> future;
92    private final ReloadingReference<PeriodicRunnerState> cachedState;
93  
94    public PeriodicRunner() {
95  
96      this.config = ConfigUtils.getDefaultConfig();
97  
98      lastUpdateTime = System.currentTimeMillis();
99      int threshold = config.getInt("model.time-threshold");
100     if (threshold < 0) {
101       nextScheduledRunTime = Long.MAX_VALUE;
102     } else {
103       nextScheduledRunTime = lastUpdateTime + TimeUnit.MILLISECONDS.convert(threshold, TimeUnit.MINUTES);
104     }
105 
106     forceRun = false;
107     running = false;
108 
109     //this.allGenerationRunners = Lists.newCopyOnWriteArrayList();
110     this.allGenerationRunners = new CopyOnWriteArrayList<GenerationRunner>();
111 
112     executor = Executors.newSingleThreadScheduledExecutor(
113         new ThreadFactoryBuilder().setNameFormat("PeriodicRunner-%d").build());
114     future = executor.scheduleWithFixedDelay(this, 0L, CHECK_INTERVAL_MINS, TimeUnit.MINUTES);
115 
116     this.cachedState = new ReloadingReference<PeriodicRunnerState>(new Callable<PeriodicRunnerState>() {
117       @Override
118       public PeriodicRunnerState call() throws IOException, InterruptedException {
119         return doGetState();
120       }
121     }, 15, TimeUnit.SECONDS);
122   }
123 
124   public PeriodicRunnerState getState() {
125     return cachedState.get(3, TimeUnit.SECONDS);
126   }
127 
128   private PeriodicRunnerState doGetState() throws IOException, InterruptedException {
129 
130     List<GenerationRunnerState> states = Lists.newArrayListWithCapacity(allGenerationRunners.size());
131     for (GenerationRunner runner : allGenerationRunners) {
132       GenerationRunnerState runnerSate = runner.getState();
133       if (runnerSate != null) {
134         states.add(runnerSate);
135       }
136     }
137     Collections.reverse(states);
138     Date nextScheduledRunTimeDate = nextScheduledRunTime == Long.MAX_VALUE ? null : new Date(nextScheduledRunTime);
139 
140     List<String> mostRecentGenerations = StoreUtils.listGenerationsForInstance(config.getString("model.instance-dir"));
141     int currentGenerationMB = dataInCurrentGenerationMB(mostRecentGenerations);
142 
143     return new PeriodicRunnerState(states, running, nextScheduledRunTimeDate, currentGenerationMB);
144   }
145 
146   public void forceRun() {
147     forceRun = true;
148     future.cancel(true);
149     future = executor.scheduleWithFixedDelay(this, 0L, CHECK_INTERVAL_MINS, TimeUnit.MINUTES);
150   }
151 
152   public void interrupt() {
153     future.cancel(true);
154     future = executor.scheduleWithFixedDelay(this, CHECK_INTERVAL_MINS, CHECK_INTERVAL_MINS, TimeUnit.MINUTES);
155   }
156 
157   @Override
158   public void close() {
159     ExecutorUtils.shutdownNowAndAwait(executor);
160     future.cancel(true);
161   }
162 
163   @Override
164   public void run()  {
165 
166     try {
167       running = true;
168       long now = System.currentTimeMillis();
169 
170       boolean timeThresholdExceeded = now >= nextScheduledRunTime;
171 
172       String instanceDir = config.getString("model.instance-dir");
173       List<String> mostRecentGenerations = StoreUtils.listGenerationsForInstance(instanceDir);
174 
175       if (!isAnyCompleteGeneration(instanceDir, mostRecentGenerations)) {
176         log.info("Forcing run -- no complete generations yet");
177         forceRun = true;
178       }
179 
180       int currentGenerationMB = dataInCurrentGenerationMB(mostRecentGenerations);
181       int dataThresholdMB = config.getInt("model.data-threshold");
182       boolean dataThresholdExceeded =
183           dataThresholdMB > 0 && currentGenerationMB >= dataThresholdMB;
184 
185       if (timeThresholdExceeded) {
186         log.info("Running new generation due to elapsed time: {} minutes",
187                  TimeUnit.MINUTES.convert(now - lastUpdateTime, TimeUnit.MILLISECONDS));
188       }
189       if (dataThresholdExceeded) {
190         log.info("Running new generation due to data written: {}MB", currentGenerationMB);
191       }
192 
193       if (forceRun || timeThresholdExceeded || dataThresholdExceeded) {
194         forceRun = false;
195         GenerationRunner newGenerationRunner = buildGenerationRunner();
196         allGenerationRunners.add(newGenerationRunner);
197 
198         // If time threshold was exceeded, pretend we started the run at the scheduled time. We noticed
199         // up to a minute later, but conceptually it started then. Otherwise the start time is
200         // conceptually now (due to data)
201         if (now > nextScheduledRunTime) {
202           lastUpdateTime = nextScheduledRunTime;
203         } else {
204           lastUpdateTime = now;
205         }
206         int minBetweenRuns = config.getInt("model.time-threshold");
207         if (minBetweenRuns < 0) {
208           nextScheduledRunTime = Long.MAX_VALUE;
209         } else {
210           long msBetweenRuns = TimeUnit.MILLISECONDS.convert(minBetweenRuns, TimeUnit.MINUTES);
211           nextScheduledRunTime = lastUpdateTime + msBetweenRuns;
212         }
213 
214         newGenerationRunner.call();
215 
216         // If we ran over time limit, schedule will run again immediately. For bookkeeping correctness,
217         // move back the scheduled run time to now, to reflect the fact that it was preempted.
218         long endTime = System.currentTimeMillis();
219         if (endTime > nextScheduledRunTime) {
220           nextScheduledRunTime = endTime;
221         }
222       }
223 
224     } catch (InterruptedException ignored) {
225       log.warn("Interrupted");
226     } catch (IOException ioe) {
227       log.warn("Unexpected error in execution", ioe);
228     } catch (JobException je) {
229       log.warn("Unexpected error in execution", je);
230     } catch (Throwable t) {
231       log.error("Unexpected error in execution", t);
232     } finally {
233       running = false;
234     }
235   }

237   private static boolean isAnyCompleteGeneration(String instanceDir,
238                                                  Collection<String> mostRecentGenerations) throws IOException {
239     if (mostRecentGenerations.isEmpty()) {
240       return false;
241     }
242     Store store = Store.get();
243     for (CharSequence recentGenerationPathString : mostRecentGenerations) {
244       int generationID = StoreUtils.parseGenerationFromPrefix(recentGenerationPathString);
245       if (store.exists(Namespaces.getGenerationDoneKey(instanceDir, generationID), true)) {
246         return true;
247       }
248     }
249     return false;
250   }
251 
252   private int dataInCurrentGenerationMB(List<String> mostRecentGenerations) throws IOException {
253     if (mostRecentGenerations.isEmpty()) {
254       return 0;
255     }
256     int currentGenerationID =
257         StoreUtils.parseGenerationFromPrefix(mostRecentGenerations.get(mostRecentGenerations.size() - 1));
258     String inboundPrefix =
259         Namespaces.getInstanceGenerationPrefix(config.getString("model.instance-dir"), currentGenerationID) +
260             "inbound/";
261 
262     long totalSizeBytes = Store.get().getSizeRecursive(inboundPrefix);
263     return (int) (totalSizeBytes / 1000000);
264   }
265 
266   private GenerationRunner buildGenerationRunner() {
267     List<Class<? extends GenerationRunner>> localDistributed = RUNNERS.get(config.getString("model.type"));
268     Preconditions.checkNotNull(localDistributed,
269                                "Unspecified or unsupported model type: %s",
270                                config.getString("model.type"));
271     Config config = ConfigUtils.getDefaultConfig();
272     boolean localComputation;
273     if (config.hasPath("model.local")) {
274       log.warn("model.local is deprecated; use model.local-computation");
275       localComputation = config.getBoolean("model.local");
276     } else {
277       localComputation = config.getBoolean("model.local-computation");
278     }
279     Class<? extends GenerationRunner> runnerClass =
280         localComputation ? localDistributed.get(0) : localDistributed.get(1);
281     return ClassUtils.loadInstanceOf(runnerClass);
282   }
