65  /**
66   * <p>This class will periodically run one generation of the Computation Layer using
67   * {@link com.cloudera.oryx.computation.common.DistributedGenerationRunner}. It can run after a period of time has elapsed, or an amount of
68   * data has been written.</p>
69   *
70   * <p>Example:</p>
71   *
72   * <p>{@code java -Dconfig.file=[config.conf] -jar oryx-computation-x.y.jar}</p>
73   *
74   * @author Sean Owen
75   */
76  public final class Runner implements Callable<Object>, Closeable {
77  
78    private static final Logger log = LoggerFactory.getLogger(Runner.class);
79  
80    private static final int[] ERROR_PAGE_STATUSES = {
81        HttpServletResponse.SC_BAD_REQUEST,
82        HttpServletResponse.SC_UNAUTHORIZED,
83        HttpServletResponse.SC_NOT_FOUND,
84        HttpServletResponse.SC_METHOD_NOT_ALLOWED,
85        HttpServletResponse.SC_INTERNAL_SERVER_ERROR,
86        HttpServletResponse.SC_SERVICE_UNAVAILABLE,
87    };
88  
89    private final Config config;
90    private Tomcat tomcat;
91    private File noSuchBaseDir;
92    private boolean closed;
93  
94    public Runner() {
95      this.config = ConfigUtils.getDefaultConfig();
96    }
97  
98    /**
99     * Starts the main loop, which runs indefinitely.
100    */
101   @Override
102   public synchronized Object call() throws IOException {
103 
104     if (closed) {
105       return Boolean.FALSE;
106     }
107 
108     MemoryHandler.setSensibleLogFormat();
109     java.util.logging.Logger.getLogger("").addHandler(new MemoryHandler());
110 
111     System.setProperty("org.apache.tomcat.util.buf.UDecoder.ALLOW_ENCODED_SLASH", "true");
112 
113     this.noSuchBaseDir = Files.createTempDir();
114     this.noSuchBaseDir.deleteOnExit();
115 
116     Tomcat tomcat = new Tomcat();
117     Connector connector = makeConnector();
118     configureTomcat(tomcat, connector);
119     configureEngine(tomcat.getEngine());
120     configureServer(tomcat.getServer());
121     configureHost(tomcat.getHost());
122     Context context = makeContext(tomcat, noSuchBaseDir);
123 
124     context.addApplicationListener(new ApplicationListener(ComputationInitListener.class.getName(), false));
125 
126     addServlet(context, new PeriodicRunnerServlet(), "/periodicRunner/*");
127     addServlet(context, new style_jspx(), "/style.jspx");
128     addServlet(context, new index_jspx(), "/index.jspx");
129     addServlet(context, new status_jspx(), "/status.jspx");
130     addServlet(context, new error_jspx(), "/error.jspx");
131     addServlet(context, new LogServlet(), "/log.txt");
132 
133     try {
134       tomcat.start();
135     } catch (LifecycleException le) {
136       throw new IOException(le);
137     }
138     this.tomcat = tomcat;
139 
140     return null;
141   }
142 
143   /**
144    * Blocks and waits until the server shuts down.
145    */
146   void await() {
147     Server server = null;
148     synchronized (this) {
149       if (tomcat != null) {
150         server = tomcat.getServer();
151       }
152     }
153     if (server != null) {
154       server.await();
155     }
156   }
157 
158   @Override
159   public synchronized void close() {
160     if (!closed) {
161       closed = true;
162       if (tomcat != null) {
163         try {
164           tomcat.stop();
165           tomcat.destroy();
166         } catch (LifecycleException le) {
167           log.warn("Unexpected error while stopping", le);
168         }
169         try {
170           IOUtils.deleteRecursively(noSuchBaseDir);
171         } catch (IOException e) {
172           log.warn("Failed to delete {}", noSuchBaseDir);
173         }
174       }
175     }
176   }
177 
178   public static void main(String[] args) throws Exception {
179     final Runner runner = new Runner();
180     SignalManager.register(new Runnable() {
181       @Override
182       public void run() {
183         runner.close();
184       }
185     }, SignalType.INT, SignalType.TERM);
186     try {
187       runner.call();
188       runner.await();
189     } finally {
190       runner.close();
191     }
192   }

194   private void configureTomcat(Tomcat tomcat, Connector connector) {
195     tomcat.setBaseDir(noSuchBaseDir.getAbsolutePath());
196     tomcat.setConnector(connector);
197     tomcat.getService().addConnector(connector);
198   }

200   private void configureEngine(Engine engine) {
201     APISettings apiSettings = APISettings.create(config.getConfig("computation-layer.api"));
202     String userName = apiSettings.getUserName();
203     String password = apiSettings.getPassword();
204     if (userName != null && password != null) {
205       InMemoryRealm realm = new InMemoryRealm();
206       realm.addUser(userName, password);
207       engine.setRealm(realm);
208     }
209   }
210 
211   private static void configureServer(Server server) {
212     server.addLifecycleListener(new JreMemoryLeakPreventionListener());
213     server.addLifecycleListener(new ThreadLocalLeakPreventionListener());
214   }
215 
216   private static void configureHost(Host host) {
217     host.setAutoDeploy(false);
218   }

220   private Connector makeConnector() {
221     Connector connector = new Connector("org.apache.coyote.http11.Http11NioProtocol");
222     APISettings apiSettings = APISettings.create(config.getConfig("computation-layer.api"));
223     File keystoreFile = apiSettings.getKeystoreFile();
224     if (keystoreFile == null) {
225       // HTTP connector
226       connector.setPort(apiSettings.getPort());
227       connector.setSecure(false);
228       connector.setScheme("http");
229 
230     } else {
231       // HTTPS connector
232       connector.setPort(apiSettings.getSecurePort());
233       connector.setSecure(true);
234       connector.setScheme("https");
235       connector.setAttribute("SSLEnabled", "true");
236       String protocol = chooseSSLProtocol("TLSv1.1", "TLSv1");
237       if (protocol != null) {
238         connector.setAttribute("sslProtocol", protocol);
239       }
240       connector.setAttribute("keystoreFile", keystoreFile.getAbsoluteFile());
241       connector.setAttribute("keystorePass", apiSettings.getKeystorePassword());
242     }
243 
244     // Keep quiet about the server type
245     connector.setXpoweredBy(false);
246     connector.setAttribute("server", "Oryx");
247 
248     return connector;
249   }
250   
251   private static String chooseSSLProtocol(String... protocols) {
252     for (String protocol : protocols) {
253       try {
254         SSLContext.getInstance(protocol);
255         return protocol;
256       } catch (NoSuchAlgorithmException ignored) {
257         // continue
258       }
259     }
260     return null;
261   }

263   private Context makeContext(Tomcat tomcat, File noSuchBaseDir) throws IOException {
264 
265     File contextPath = new File(noSuchBaseDir, "context");
266     IOUtils.mkdirs(contextPath);
267 
268     Context context = tomcat.addContext("", contextPath.getAbsolutePath());
269     context.setWebappVersion("3.0");
270     context.addWelcomeFile("index.jspx");
271     addErrorPages(context);
272 
273     context.addServletContainerInitializer(new JasperInitializer(), Collections.<Class<?>>emptySet());
274 
275     APISettings apiSettings = APISettings.create(config.getConfig("computation-layer.api"));
276 
277     boolean needHTTPS = apiSettings.isSecure();
278     boolean needAuthentication = apiSettings.getUserName() != null;
279 
280     if (needHTTPS || needAuthentication) {
281 
282       SecurityCollection securityCollection = new SecurityCollection();
283       securityCollection.addPattern("/*");
284       SecurityConstraint securityConstraint = new SecurityConstraint();
285       securityConstraint.addCollection(securityCollection);
286 
287       if (needHTTPS) {
288         securityConstraint.setUserConstraint("CONFIDENTIAL");
289       }
290 
291       if (needAuthentication) {
292 
293         LoginConfig loginConfig = new LoginConfig();
294         loginConfig.setAuthMethod("DIGEST");
295         loginConfig.setRealmName(InMemoryRealm.NAME);
296         context.setLoginConfig(loginConfig);
297 
298         securityConstraint.addAuthRole(InMemoryRealm.AUTH_ROLE);
299 
300         context.addSecurityRole(InMemoryRealm.AUTH_ROLE);
301         context.getPipeline().addValve(new DigestAuthenticator());
302       }
303 
304       context.addConstraint(securityConstraint);
305     }
306 
307     context.setCookies(false);
308 
309     return context;
310   }

312   private static void addServlet(Context context, Servlet servlet, String path) {
313     String name = servlet.getClass().getSimpleName();
314     Tomcat.addServlet(context, name, servlet).setLoadOnStartup(1);
315     context.addServletMapping(path, name);
316   }

318   private static void addErrorPages(Context context) {
319     for (int errorCode : ERROR_PAGE_STATUSES) {
320       ErrorPage errorPage = new ErrorPage();
321       errorPage.setErrorCode(errorCode);
322       errorPage.setLocation("/error.jspx");
323       context.addErrorPage(errorPage);
324     }
325     ErrorPage errorPage = new ErrorPage();
326     errorPage.setExceptionType(Throwable.class.getName());
327     errorPage.setLocation("/error.jspx");
328     context.addErrorPage(errorPage);
329   }
