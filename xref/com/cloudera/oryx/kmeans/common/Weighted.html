16  package com.cloudera.oryx.kmeans.common;
17  
18  import java.util.Collection;
19  import java.util.List;
20  import java.util.SortedMap;
21  
22  import com.google.common.base.Function;
23  import com.google.common.base.Preconditions;
24  import com.google.common.collect.Collections2;
25  import com.google.common.collect.Lists;
26  import com.google.common.collect.Maps;
27  import org.apache.commons.math3.random.RandomGenerator;
28  
29  import com.cloudera.oryx.common.LangUtils;
30  import com.cloudera.oryx.common.random.RandomManager;
31  
32  /**
33   * A typed container for an object and an associated numeric weight.
34   */
35  public class Weighted<T> {
36  
37    private final T thing;
38    private final double weight;
39    
40    private static final class WeightFunction<T> implements Function<T, Weighted<T>> {
41      @Override
42      public Weighted<T> apply(T input) {
43        return new Weighted<T>(input);
44      }
45    }
46    
47    /**
48     * Converts an input {@code Collection<T>} into a {@code Collection<Weighted>}.
49     * 
50     * @param things The items to convert
51     * @return The items as {@code Weighted<T>} values with weight 1.0
52     */
53    public static <T> Collection<Weighted<T>> create(Collection<T> things) {
54      return Collections2.transform(things, new WeightFunction<T>());
55    }
56    
57    /**
58     * Converts an input {@code List<T>} into a {@code List<Weighted>}.
59     * 
60     * @param things The items to convert
61     * @return The items as {@code Weighted<T>} values with weight 1.0
62     */
63    public static <T> List<Weighted<T>> create(List<T> things) {
64      return Lists.transform(things, new WeightFunction<T>());
65    }
66    
67    /**
68     * Sample items from a {@code List<Weighted>} where items with higher weights
69     * have a higher probability of being included in the sample.
70     * 
71     * @param things The iterable to sample from
72     * @param size The number of items to sample
73     * @return A list containing the sampled items
74     */
75    public static <T extends Weighted<?>> List<T> sample(Iterable<T> things, int size, RandomGenerator random) {
76      if (random == null) {
77        random = RandomManager.getRandom();
78      }
79      SortedMap<Double, T> sampled = Maps.newTreeMap();
80      for (T thing : things) {
81        if (thing.weight() > 0) {
82          double score = Math.log(random.nextDouble()) / thing.weight();
83          if (sampled.size() < size || score > sampled.firstKey()) {
84            sampled.put(score, thing);
85          }
86          if (sampled.size() > size) {
87            sampled.remove(sampled.firstKey());
88          }
89        }
90      }
91      return Lists.newArrayList(sampled.values());
92    }
93  
94    /**
95     * Create a new instance with weight 1.0.
96     * 
97     * @param thing The thing that is weighted
98     */
99    public Weighted(T thing) {
100     this(thing, 1.0);
101   }
102   
103   /**
104    * Create a new instance with the given weight.
105    * 
106    * @param thing The (non-null) thing to weight
107    * @param weight The weight
108    */
109   public Weighted(T thing, double weight) {
110     this.thing = Preconditions.checkNotNull(thing);
111     this.weight = weight;
112   }
113   
114   /**
115    * Return the thing referenced by this instance.
116    */
117   public T thing() {
118     return thing;
119   }
120   
121   /**
122    * Return the numeric weight for this instance.
123    */
124   public double weight() {
125     return weight;
126   }
127   
128   @Override
129   public boolean equals(Object other) {
130     if (!(other instanceof Weighted)) {
131       return false;
132     }
133     Weighted<?> wv = (Weighted<?>) other;
134     return thing.equals(wv.thing) && weight == wv.weight;
135   }
136   
137   @Override
138   public int hashCode() {
139     return 17 * thing.hashCode() + 37 * LangUtils.hashDouble(weight);
140   }
141   
142   @Override
143   public String toString() {
144     return String.valueOf(thing) + ';' + weight;
145   }
146 }
