35  
36  public final class KSketchSamplingStep extends KMeansJobStep {
37  
38    @Override
39    protected MRPipeline createPipeline() throws IOException {
40      JobStepConfig stepConfig = getConfig();
41      ClusterSettings settings = ClusterSettings.create(ConfigUtils.getDefaultConfig());
42  
43      String instanceDir = stepConfig.getInstanceDir();
44      int generationID = stepConfig.getGenerationID();
45      int iteration = stepConfig.getIteration();
46      String prefix = Namespaces.getInstanceGenerationPrefix(instanceDir, generationID);
47      String outputKey = prefix + String.format("sketch/%d/", iteration);
48      if (!validOutputPath(outputKey)) {
49        return null;
50      }
51  
52      // get normalized vectors
53      String inputKey = prefix + "normalized/";
54      MRPipeline p = createBasicPipeline(DistanceToClosestFn.class);
55      AvroType<Pair<Integer, RealVector>> inputType = Avros.pairs(Avros.ints(), MLAvros.vector());
56      PCollection<Pair<Integer, RealVector>> in = p.read(avroInput(inputKey, inputType));
57  
58      // either create or load the set of currently chosen k-sketch vectors
59      // they are stored in a KSketchIndex object
60      DistanceToClosestFn<RealVector> distanceToClosestFn;
61      UpdateIndexFn updateIndexFn;
62      if (iteration == 1) { // Iteration 1 is the first real iteration; iteration 0 contains initial state
63        KSketchIndex index = createInitialIndex(settings, in);
64        distanceToClosestFn = new DistanceToClosestFn<RealVector>(index);
65        updateIndexFn = new UpdateIndexFn(index);
66      } else {
67        // Get the index location from the previous iteration
68        String previousIndexKey = prefix + String.format("sketch/%d/", iteration - 1);
69        distanceToClosestFn = new DistanceToClosestFn<RealVector>(previousIndexKey);
70        updateIndexFn = new UpdateIndexFn(previousIndexKey);
71      }
72  
73      // compute distance of each vector in dataset to closest vector in k-sketch
74      PTable<Integer, Pair<RealVector, Double>> weighted = in.parallelDo("computeDistances", distanceToClosestFn,
75          Avros.tableOf(Avros.ints(), Avros.pairs(MLAvros.vector(), Avros.doubles())));
76  
77      // run weighted reservoir sampling on the vector to select another group of settings.getSketchPoints()
78      // to add to the k-sketch
79      PTable<Integer,RealVector> kSketchSample = ReservoirSampling.groupedWeightedSample(weighted,
80          settings.getSketchPoints(), RandomManager.getRandom());
81  
82      // update the KSketchIndex with the newly-chosen vectors
83      kSketchSample.parallelDo("updateIndex", updateIndexFn, Serializables.avro(KSketchIndex.class))
84          .write(avroOutput(outputKey));
85  
86      return p;
87    }
88  
89    public static KSketchIndex createInitialIndex(ClusterSettings settings,
90                                                   PCollection<Pair<Integer, RealVector>> input) {
91      RealVector[] init = new RealVector[settings.getCrossFolds()];
92      for (Pair<Integer, RealVector> rv : input.materialize()) {
93        if (init[rv.first()] == null) {
94          init[rv.first()] = rv.second();
95        }
96        boolean done = true;
97        for (RealVector vec : init) {
98          if (vec == null) {
99            done = false;
100           break;
101         }
102       }
103       if (done) {
104         break;
105       }
106     }
107 
108     KSketchIndex index = new KSketchIndex(
109         settings.getCrossFolds(),
110         init[0].getDimension(),
111         settings.getIndexBits(),
112         settings.getIndexSamples(),
113         1729L); // TODO: something smarter, or figure out that I don't need this b/c I compute the projections up front
114     for (int i = 0; i < init.length; i++) {
115       index.add(init[i], i);
116     }
117     return index;
118   }
119 }
