16  package com.cloudera.oryx.kmeans.computation.local;
17  
18  import com.cloudera.oryx.common.io.DelimitedDataUtils;
19  import com.cloudera.oryx.common.io.IOUtils;
20  import com.cloudera.oryx.common.iterator.FileLineIterable;
21  import com.cloudera.oryx.common.math.NamedRealVector;
22  import com.cloudera.oryx.common.math.Vectors;
23  import com.cloudera.oryx.common.settings.ConfigUtils;
24  import com.cloudera.oryx.common.settings.InboundSettings;
25  import com.cloudera.oryx.computation.common.crossfold.Crossfold;
26  import com.cloudera.oryx.computation.common.summary.Summary;
27  import com.cloudera.oryx.computation.common.summary.SummaryStats;
28  import com.cloudera.oryx.kmeans.computation.normalize.NormalizeSettings;
29  import com.cloudera.oryx.kmeans.computation.normalize.Transform;
30  import com.google.common.collect.Lists;
31  import com.typesafe.config.Config;
32  import org.apache.commons.math3.linear.RealVector;
33  import org.apache.commons.math3.random.RandomGenerator;
34  import org.slf4j.Logger;
35  import org.slf4j.LoggerFactory;
36  
37  import java.io.File;
38  import java.io.IOException;
39  import java.util.Collection;
40  import java.util.List;
41  import java.util.concurrent.Callable;
42  
43  public final class Standardize implements Callable<List<List<RealVector>>> {
44  
45    private static final Logger log = LoggerFactory.getLogger(Standardize.class);
46  
47    private final File inputDir;
48    private final Summary summary;
49  
50    public Standardize(File inputDir, Summary summary) {
51      this.inputDir = inputDir;
52      this.summary = summary;
53    }
54  
55    @Override
56    public List<List<RealVector>> call() throws IOException {
57      File[] inputFiles = inputDir.listFiles(IOUtils.NOT_HIDDEN);
58      if (inputFiles == null || inputFiles.length == 0) {
59        log.info("No input files in {}", inputDir);
60        return null;
61      }
62  
63      Config config = ConfigUtils.getDefaultConfig();
64      InboundSettings inboundSettings = InboundSettings.create(config);
65      int numFeatures = inboundSettings.getColumnNames().size();
66      NormalizeSettings settings = NormalizeSettings.create(config);
67      Crossfold crossfold = new Crossfold(config.getInt("model.cross-folds"));
68      RandomGenerator rand = crossfold.getRandomGenerator();
69  
70      Collection<Integer> ignoredColumns = inboundSettings.getIgnoredColumns();
71      Collection<Integer> idColumns = inboundSettings.getIdColumns();
72      int expansion = -ignoredColumns.size() + summary.getNetLevels() -
73          (!idColumns.isEmpty() && !ignoredColumns.contains(idColumns.iterator().next()) ? 1 : 0);
74      boolean sparse = settings.getSparse() != null ? settings.getSparse() :
75          expansion > 2 * (summary.getFieldCount() - ignoredColumns.size());
76  
77      List<List<RealVector>> ret = Lists.newArrayList();
78      for (int i = 0; i < crossfold.getNumFolds(); i++) {
79        ret.add(Lists.<RealVector>newLinkedList());
80      }
81      for (File inputFile : inputFiles) {
82        log.info("Standardizing input from {}", inputFile.getName());
83        for (String line : new FileLineIterable(inputFile)) {
84          if (line.isEmpty()) {
85            continue;
86          }
87          String[] tokens = DelimitedDataUtils.decode(line);
88          RealVector v = sparse ? Vectors.sparse(tokens.length + expansion) : Vectors.dense(tokens.length + expansion);
89          int offset = 0;
90          for (int i = 0; i < numFeatures; i++) {
91            if (inboundSettings.isIgnored(i)) {
92              // Do nothing
93            } else if (inboundSettings.isNumeric(i)) {
94              SummaryStats ss = summary.getStats(i);
95              Transform t = settings.getTransform(i);
96              double raw = asDouble(tokens[i]);
97              if (!Double.isNaN(raw)) {
98                double n = t.apply(raw, ss) * settings.getScale(i);
99                v.setEntry(offset, n);
100             }
101             offset++;
102           } else if (inboundSettings.isCategorical(i)) {
103             SummaryStats ss = summary.getStats(i);
104             int index = ss.index(tokens[i]);
105             if (index >= 0) {
106               v.setEntry(offset + index, settings.getScale(i));
107               offset += ss.numLevels();
108             } else {
109               log.warn("Unrecognized value for category {}: {}", i, tokens[i]);
110             }
111           }
112         }
113         if (!inboundSettings.getIdColumns().isEmpty()) {
114           // TODO: multiple ID columns
115           v = new NamedRealVector(v, tokens[inboundSettings.getIdColumns().iterator().next()]);
116         }
117         // Assign the vector to a fold
118         int fold = rand.nextInt(crossfold.getNumFolds());
119         ret.get(fold).add(v);
120       }
121     }
122     return ret;
123   }
124 
125   private static double asDouble(String token) {
126     try {
127       return Double.valueOf(token);
128     } catch (NumberFormatException e) {
129       log.warn("Invalid numeric token: {}", token);
130       return Double.NaN;
131     }
132   }
133 }
