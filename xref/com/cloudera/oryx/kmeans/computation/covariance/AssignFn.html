15  package com.cloudera.oryx.kmeans.computation.covariance;
16  
17  import com.cloudera.oryx.computation.common.fn.OryxDoFn;
18  import com.cloudera.oryx.kmeans.common.Distance;
19  import com.cloudera.oryx.kmeans.computation.cluster.KSketchIndex;
20  import org.apache.commons.math3.linear.RealVector;
21  import org.apache.crunch.Emitter;
22  import org.apache.crunch.Pair;
23  
24  public final class AssignFn<V extends RealVector> extends OryxDoFn<V, Pair<ClusterKey, Pair<V, Double>>> {
25    private final KSketchIndex centers;
26    private final boolean approx;
27  
28    public AssignFn(KSketchIndex centers, boolean approx) {
29      this.centers = centers;
30      this.approx = approx;
31    }
32  
33    @Override
34    public void process(V vec, Emitter<Pair<ClusterKey, Pair<V, Double>>> emitter) {
35      Distance[] distances = centers.getDistances(vec, approx);
36      int[] pointCounts = centers.getPointCounts();
37      for (int i = 0; i < distances.length; i++) {
38        ClusterKey key = new ClusterKey(pointCounts[i], distances[i].getClosestCenterId());
39        emitter.emit(Pair.of(key, Pair.of(vec, distances[i].getSquaredDistance())));
40      }
41    }
42  }
