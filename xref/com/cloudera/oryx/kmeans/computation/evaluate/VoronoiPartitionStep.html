26  import org.apache.commons.math3.linear.RealVector;
27  import org.apache.crunch.PCollection;
28  import org.apache.crunch.impl.mr.MRPipeline;
29  import org.apache.crunch.lib.PTables;
30  import org.apache.crunch.types.avro.Avros;
31  
32  import java.io.IOException;
33  
34  public final class VoronoiPartitionStep extends KMeansJobStep {
35  
36    @Override
37    protected MRPipeline createPipeline() throws IOException {
38      JobStepConfig stepConfig = getConfig();
39      Config config = ConfigUtils.getDefaultConfig();
40      ClusterSettings clusterSettings = ClusterSettings.create(config);
41  
42      String instanceDir = stepConfig.getInstanceDir();
43      int generationID = stepConfig.getGenerationID();
44      String prefix = Namespaces.getInstanceGenerationPrefix(instanceDir, generationID);
45      String outputKey = prefix + "weighted/";
46      if (!validOutputPath(outputKey)) {
47        return null;
48      }
49  
50      String indexKey = prefix + "sketch/" + clusterSettings.getSketchIterations();
51      String inputKey = prefix + "normalized/";
52      MRPipeline p = createBasicPipeline(ClosestSketchVectorFn.class);
53  
54      // first I compute the weight of each k-sketch vector, i.e., Voronoi partition
55      // I aggregate all together and persist on disk
56      // PCollection<ClosestSketchVectorData> weights = inputPairs(p, inputKey, MLAvros.vector())
57      PCollection<ClosestSketchVectorData> weights = PTables.asPTable(inputPairs(p, inputKey, MLAvros.vector())
58          .parallelDo("computingSketchVectorWeights",
59              new ClosestSketchVectorFn<RealVector>(indexKey, clusterSettings),
60              Avros.pairs(Avros.ints(), Avros.reflects(ClosestSketchVectorData.class))))
61          .groupByKey(1)
62          .combineValues(new ClosestSketchVectorAggregator(clusterSettings))
63          .values()
64          .write(avroOutput(outputKey + "kSketchVectorWeights/"));
65  
66      // this "pipeline" takes a single ClosestSketchVectorData and returns weighted vectors
67      // could be done outside MapReduce, but that would require me to materialize the ClosestSketchVectorData
68      weights.parallelDo(
69          "generatingWeightedSketchVectors",
70          new WeightVectorsFn(indexKey),
71          KMeansTypes.FOLD_WEIGHTED_VECTOR)
72          .write(avroOutput(outputKey + "weightedKSketchVectors/"));
73  
74      return p;
75    }
76  }
