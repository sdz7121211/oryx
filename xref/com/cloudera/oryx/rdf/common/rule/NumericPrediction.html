16  package com.cloudera.oryx.rdf.common.rule;
17  
18  import com.google.common.base.Preconditions;
19  import org.apache.commons.math3.stat.descriptive.StorelessUnivariateStatistic;
20  import org.apache.commons.math3.stat.descriptive.moment.Mean;
21  
22  import com.cloudera.oryx.rdf.common.example.Example;
23  import com.cloudera.oryx.rdf.common.example.FeatureType;
24  import com.cloudera.oryx.rdf.common.example.NumericFeature;
25  
26  /**
27   * Represents a predicted value of a numeric target. The prediction is simply a real number in this case.
28   *
29   * @author Sean Owen
30   * @see CategoricalPrediction
31   */
32  public final class NumericPrediction extends Prediction {
33  
34    private float prediction;
35  
36    public NumericPrediction(float prediction, int initialCount) {
37      super(initialCount);
38      this.prediction = prediction;
39    }
40  
41    public float getPrediction() {
42      return prediction;
43    }
44  
45    @Override
46    public FeatureType getFeatureType() {
47      return FeatureType.NUMERIC;
48    }
49  
50    @Override
51    public synchronized void update(Example train) {
52      NumericFeature target = (NumericFeature) train.getTarget();
53      int oldCount = getCount();
54      int newCount = oldCount + 1;
55      setCount(newCount);
56      prediction =
57          (float) ((oldCount / (double) newCount) * prediction +
58                   target.getValue() / (double) newCount);
59    }
60  
61    @Override
62    public boolean equals(Object o) {
63      if (!(o instanceof NumericPrediction)) {
64        return false;
65      }
66      NumericPrediction other = (NumericPrediction) o;
67      return prediction == other.prediction;
68    }
69  
70    @Override
71    public int hashCode() {
72      return Float.floatToIntBits(prediction);
73    }
74  
75    @Override
76    public String toString() {
77      return Float.toString(prediction);
78    }
79  
80    static NumericPrediction buildNumericPrediction(Iterable<Example> examples) {
81      StorelessUnivariateStatistic mean = new Mean();
82      for (Example example : examples) {
83        mean.increment(((NumericFeature) example.getTarget()).getValue());
84      }
85      Preconditions.checkState(mean.getN() > 0);
86      return new NumericPrediction((float) mean.getResult(), (int) mean.getN());
87    }
88  
89  }
