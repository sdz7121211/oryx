16  package com.cloudera.oryx.rdf.common.rule;
17  
18  import com.google.common.collect.Lists;
19  import com.google.common.collect.Multiset;
20  import com.google.common.collect.TreeMultiset;
21  import org.apache.commons.math3.stat.descriptive.StorelessUnivariateStatistic;
22  import org.apache.commons.math3.stat.descriptive.moment.Mean;
23  import org.apache.commons.math3.util.FastMath;
24  
25  import java.util.Collections;
26  import java.util.List;
27  
28  import com.cloudera.oryx.rdf.common.example.Example;
29  import com.cloudera.oryx.rdf.common.example.FeatureType;
30  import com.cloudera.oryx.rdf.common.example.NumericFeature;
31  
32  /**
33   * Represents a decision over a numeric feature. Decisions are defined simply by a threshold value; the
34   * decision is positive if the feature's value is greater than or equal to this decision's threshold value.
35   *
36   * @author Sean Owen
37   * @see CategoricalDecision
38   */
39  public final class NumericDecision extends Decision {
40  
41    private final float threshold;
42    private final boolean defaultDecision;
43  
44    public NumericDecision(int featureNumber, float threshold, boolean defaultDecision) {
45      super(featureNumber);
46      this.threshold = threshold;
47      this.defaultDecision = defaultDecision;
48    }
49  
50    private NumericDecision(int featureNumber, float threshold, float mean) {
51      super(featureNumber);
52      this.threshold = threshold;
53      this.defaultDecision = mean >= threshold;
54    }
55  
56    /**
57     * @return decision threshold; feature values greater than or equal are considered positive
58     */
59    public float getThreshold() {
60      return threshold;
61    }
62  
63    @Override
64    public boolean getDefaultDecision() {
65      return defaultDecision;
66    }
67  
68    @Override
69    public boolean isPositive(Example example) {
70      NumericFeature feature = (NumericFeature) example.getFeature(getFeatureNumber());
71      return feature == null ? defaultDecision : feature.getValue() >= threshold;
72    }
73  
74    @Override
75    public FeatureType getType() {
76      return FeatureType.NUMERIC;
77    }
78  
79    @Override
80    public boolean equals(Object o) {
81      if (!(o instanceof NumericDecision)) {
82        return false;
83      }
84      NumericDecision other = (NumericDecision) o;
85      return getFeatureNumber() == other.getFeatureNumber() && threshold == other.threshold;
86    }
87  
88    @Override
89    public int hashCode() {
90      return getFeatureNumber() ^ Float.floatToIntBits(threshold);
91    }
92  
93    @Override
94    public String toString() {
95      return "(#" + getFeatureNumber() + " >= " + threshold + ')';
96    }
97  
98    static List<Decision> numericDecisionsFromExamples(int featureNumber,
99                                                       Iterable<Example> examples,
100                                                      int suggestedMaxSplitCandidates) {
101     Multiset<Float> sortedFeatureValueCounts = TreeMultiset.create();
102     StorelessUnivariateStatistic mean = new Mean();
103     int numExamples = 0;
104     for (Example example : examples) {
105       NumericFeature feature = (NumericFeature) example.getFeature(featureNumber);
106       if (feature == null) {
107         continue;
108       }
109       numExamples++;
110       float value = feature.getValue();
111       sortedFeatureValueCounts.add(value, 1);
112       mean.increment(value);
113     }
114 
115     // Make decisions from split points that divide up input into roughly equal amounts of examples
116     List<Decision> decisions = Lists.newArrayListWithExpectedSize(suggestedMaxSplitCandidates);
117     int approxExamplesPerSplit = FastMath.max(1, numExamples / suggestedMaxSplitCandidates);
118     int examplesInSplit = 0;
119     float lastValue = Float.NaN;
120     // This will iterate in order of value by nature of TreeMap
121     for (Multiset.Entry<Float> entry : sortedFeatureValueCounts.entrySet()) {
122       float value = entry.getElement();
123       if (examplesInSplit >= approxExamplesPerSplit) {
124         decisions.add(new NumericDecision(featureNumber, (value + lastValue) / 2.0f, (float) mean.getResult()));
125         examplesInSplit = 0;
126       }
127       examplesInSplit += entry.getCount();
128       lastValue = value;
129     }
130 
131     // The vital condition here is that if decision n decides an example is positive, then all subsequent
132     // decisions in the list will also find it positive. So we need to order from highest threshold to lowest
133     Collections.reverse(decisions);
134     return decisions;
135   }
136 
137 }
