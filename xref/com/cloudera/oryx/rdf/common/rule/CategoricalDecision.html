38  /**
39   * Represents a decision over a categorical feature. If the categorical feature takes on values from the
40   * set V = { v1, v2, ... } then the rule is defined by a subset of V. The decision evaluates as positive if
41   * an example's value for this feature is in the rule's subset.
42   *
43   * @author Sean Owen
44   * @see NumericDecision
45   */
46  public final class CategoricalDecision extends Decision {
47    
48    private final BitSet activeCategories;
49    private final boolean defaultDecision;
50  
51    public CategoricalDecision(int featureNumber, BitSet activeCategories, boolean defaultDecision) {
52      super(featureNumber);
53      this.activeCategories = activeCategories;
54      this.defaultDecision = defaultDecision;
55    }
56  
57    private CategoricalDecision(int featureNumber,
58                                int categoryCount,
59                                List<Pair<Double,Integer>> scoreCategoryPairs,
60                                int from,
61                                int to,
62                                int maxCategory) {
63      super(featureNumber);
64      activeCategories = new BitSet(categoryCount);
65      for (int i = from; i < to; i++) {
66        activeCategories.set(scoreCategoryPairs.get(i).getSecond());
67      }
68      defaultDecision = activeCategories.get(maxCategory);
69    }
70  
71    /**
72     * @return category value IDs which are considered positive by this decision
73     */
74    public BitSet getCategoryIDs() {
75      return activeCategories;
76    }
77  
78    @Override
79    public boolean getDefaultDecision() {
80      return defaultDecision;
81    }
82  
83    @Override
84    public boolean isPositive(Example example) {
85      CategoricalFeature feature = (CategoricalFeature) example.getFeature(getFeatureNumber());
86      return feature == null ? defaultDecision : activeCategories.get(feature.getValueID());
87    }
88  
89    @Override
90    public FeatureType getType() {
91      return FeatureType.CATEGORICAL;
92    }
93  
94    @Override
95    public boolean equals(Object o) {
96      if (!(o instanceof CategoricalDecision)) {
97        return false;
98      }
99      CategoricalDecision other = (CategoricalDecision) o;
100     return getFeatureNumber() == other.getFeatureNumber() && activeCategories.equals(other.activeCategories);
101   }
102 
103   @Override
104   public int hashCode() {
105     return getFeatureNumber() ^ activeCategories.hashCode();
106   }
107   
108   @Override
109   public String toString() {
110     StringBuilder result = new StringBuilder();
111     result.append("(#").append(getFeatureNumber()).append(" âˆˆ [");
112     int category = -1;
113     boolean first = true;
114     while ((category = activeCategories.nextSetBit(category + 1)) >= 0) {
115       if (first) {
116         first = false;
117       } else {
118         result.append(',');
119       }
120       result.append(category);
121     }
122     result.append("])");
123     return result.toString();
124   }

126   static List<Decision> categoricalDecisionsFromExamples(int featureNumber,
127                                                          ExampleSet examples,
128                                                          int suggestedMaxSplitCandidates) {
129     switch (examples.getTargetType()) {
130       case NUMERIC:
131         return categoricalDecisionsForNumericTarget(featureNumber, examples, suggestedMaxSplitCandidates);
132       case CATEGORICAL:
133         return categoricalDecisionsForCategoricalTarget(featureNumber, examples, suggestedMaxSplitCandidates);
134       default:
135         throw new IllegalStateException();
136     }
137   }
138   
139   private static List<Decision> categoricalDecisionsForNumericTarget(int featureNumber,
140                                                                      ExampleSet examples,
141                                                                      int suggestedMaxSplitCandidates) {
142     // PLANET paper claims this is optimal:
143     int categoryCount = examples.getCategoryCount(featureNumber);
144     Mean[] averageTargetForCategory = new Mean[categoryCount];
145     for (Example example : examples) {
146       CategoricalFeature feature = (CategoricalFeature) example.getFeature(featureNumber);
147       if (feature == null) {
148         continue;
149       }
150       int category = feature.getValueID();
151       Mean categoryAverage = averageTargetForCategory[category];
152       if (categoryAverage == null) {
153         categoryAverage = new Mean();
154         averageTargetForCategory[category] = categoryAverage;
155       }
156       categoryAverage.increment(((NumericFeature) example.getTarget()).getValue());
157     }
158 
159     int maxCategory = -1;
160     int maxCount = -1;
161     for (int i = 0; i < averageTargetForCategory.length; i++) {
162       Mean average = averageTargetForCategory[i];
163       if (average != null && average.getN() > maxCount) {
164         maxCount = (int) averageTargetForCategory[i].getN();
165         maxCategory = i;
166       }
167     }
168     Preconditions.checkArgument(maxCategory >= 0);
169     
170     List<Pair<Double,Integer>> byScore = Lists.newArrayListWithCapacity(averageTargetForCategory.length);
171     for (int featureCategory = 0; featureCategory < averageTargetForCategory.length; featureCategory++) {
172       StorelessUnivariateStatistic mean = averageTargetForCategory[featureCategory];
173       if (mean != null) {
174         byScore.add(new Pair<Double,Integer>(mean.getResult(), featureCategory));
175       }
176     }
177     return sortAndGetDecisionsOverSubset(featureNumber,
178                                          categoryCount,
179                                          byScore,
180                                          maxCategory,
181                                          suggestedMaxSplitCandidates);
182   }
183 
184   private static List<Decision> categoricalDecisionsForCategoricalTarget(int featureNumber,
185                                                                          ExampleSet examples,
186                                                                          int suggestedMaxSplitCandidates) {
187     int categoryCount = examples.getCategoryCount(featureNumber);
188     int[] countsForFeature = new int[categoryCount];
189     int[][] targetCountsForFeature = new int[categoryCount][];
190     for (Example example : examples) {
191       CategoricalFeature feature = (CategoricalFeature) example.getFeature(featureNumber);
192       if (feature == null) {
193         continue;
194       }
195       int featureCategory = feature.getValueID();
196       countsForFeature[featureCategory]++;
197       int[] targetCounts = targetCountsForFeature[featureCategory];
198       if (targetCounts == null) {
199         targetCounts = new int[examples.getTargetCategoryCount()];
200         targetCountsForFeature[featureCategory] = targetCounts;
201       }
202       int targetCategory = ((CategoricalFeature) example.getTarget()).getValueID();
203       targetCounts[targetCategory]++;
204     }
205 
206     int maxCategory = -1;
207     int maxCount = -1;
208     for (int i = 0; i < countsForFeature.length; i++) {
209       if (countsForFeature[i] > maxCount) {
210         maxCount = countsForFeature[i];
211         maxCategory = i;
212       }
213     }
214     Preconditions.checkArgument(maxCategory >= 0);
215 
216     List<Pair<Double,Integer>> byScore = Lists.newArrayListWithCapacity(targetCountsForFeature.length);
217     for (int featureCategory = 0; featureCategory < targetCountsForFeature.length; featureCategory++) {
218       int[] targetCategoryCounts = targetCountsForFeature[featureCategory];
219       if (targetCategoryCounts != null) {
220         double entropy = Information.entropy(targetCategoryCounts);
221         byScore.add(new Pair<Double,Integer>(entropy, featureCategory));
222       }
223     }
224 
225     return sortAndGetDecisionsOverSubset(featureNumber,
226                                          categoryCount,
227                                          byScore,
228                                          maxCategory,
229                                          suggestedMaxSplitCandidates);
230   }
231 
232   private static List<Decision> sortAndGetDecisionsOverSubset(int featureNumber,
233                                                               int categoryCount,
234                                                               List<Pair<Double,Integer>> sorted,
235                                                               int maxCategory,
236                                                               int suggestedMaxSplitCandidates) {
237     if (sorted.isEmpty()) {
238       return Collections.emptyList();
239     }
240     Collections.sort(sorted, PairComparator.INSTANCE);
241     // The vital condition here is that if decision n decides an example is positive, then all subsequent
242     // decisions in the list will also find it positive. Here we create decisions with sets of categories that
243     // are always supersets of previous ones.
244     int numDecisions = FastMath.min(sorted.size(), suggestedMaxSplitCandidates);
245     // Going to take the decisions based on earlier splits, if we need to cut it down.
246     // These are decisions that partition the data into a low-entropy subset, and everything else.
247     // These are likely to be the best.
248     List<Decision> decisions = Lists.newArrayListWithCapacity(numDecisions - 1);
249     for (int i = 1; i < numDecisions; i++) {
250       decisions.add(new CategoricalDecision(featureNumber, categoryCount, sorted, 0, i, maxCategory));
251     }
252     return decisions;
253   }
254 
255   private static final class PairComparator implements Comparator<Pair<Double,?>>, Serializable {
256     private static final Comparator<Pair<Double,?>> INSTANCE = new PairComparator();
257     @Override
258     public int compare(Pair<Double,?> a, Pair<Double,?> b) {
259       return a.getFirst().compareTo(b.getFirst());
260     }
261   }
