39  /**
40   * A decision-tree classifier. Given a set of training {@link Example}s, builds a model by randomly choosing
41   * subsets of features and the training set, and then finding a binary {@link Decision} over those features and data
42   * that produces the largest information gain in the two subsets it implies. This is repeated to build a tree
43   * of {@link DecisionNode}s. At the bottom, leaf nodes are formed ({@link TerminalNode}) that contain a
44   * {@link Prediction} of the target value.
45   *
46   * @author Sean Owen
47   * @see DecisionForest
48   */
49  public final class DecisionTree implements TreeBasedClassifier {
50  
51    private final TreeNode root;
52  
53    public static DecisionTree fromExamplesWithDefault(List<Example> examples) {
54      ExampleSet exampleSet = new ExampleSet(examples);
55      Config config = ConfigUtils.getDefaultConfig();
56      int numFeatures = exampleSet.getNumFeatures();
57      double fractionOfFeaturesToTry = config.getDouble("model.fraction-features-to-try");
58      int featuresToTry = FastMath.max(1, (int) (fractionOfFeaturesToTry * numFeatures));
59      int minNodeSize = config.getInt("model.min-node-size");
60      double minInfoGainNats = config.getDouble("model.min-info-gain-nats");
61      int suggestedMaxSplitCandidates = config.getInt("model.max-split-candidates");
62      int maxDepth = config.getInt("model.max-depth");
63      return new DecisionTree(numFeatures,
64                              featuresToTry,
65                              minNodeSize,
66                              minInfoGainNats,
67                              suggestedMaxSplitCandidates,
68                              maxDepth,
69                              exampleSet);
70    }
71  
72    public DecisionTree(int numFeatures,
73                        int featuresToTry,
74                        int minNodeSize,
75                        double minInfoGainNats,
76                        int suggestedMaxSplitCandidates,
77                        int maxDepth,
78                        ExampleSet examples) {
79      Preconditions.checkArgument(numFeatures >= 1);
80      Preconditions.checkArgument(minNodeSize >= 1);
81      Preconditions.checkArgument(minInfoGainNats >= 0.0);
82      Preconditions.checkArgument(suggestedMaxSplitCandidates >= 1);
83      Preconditions.checkArgument(maxDepth >= 1);
84      RandomGenerator random = RandomManager.getRandom();
85      root = build(examples,
86                   1,
87                   minNodeSize,
88                   minInfoGainNats,
89                   featuresToTry,
90                   numFeatures,
91                   suggestedMaxSplitCandidates,
92                   maxDepth,
93                   random);
94    }

107   private static TreeNode build(ExampleSet examples,
108                                 int buildAtDepth,
109                                 int minNodeSize,
110                                 double minInfoGainNats,
111                                 int featuresToTry,
112                                 int numFeatures,
113                                 int suggestedMaxSplitCandidates,
114                                 int maxDepth,
115                                 RandomGenerator random) {
116     if (buildAtDepth >= maxDepth - 1 || examples.getExamples().size() < minNodeSize) {
117       return new TerminalNode(Prediction.buildPrediction(examples));
118     }
119 
120     double bestGain = Double.NEGATIVE_INFINITY;
121     Decision bestDecision = null;
122 
123     for (int featureNumber : randomFeatures(examples, featuresToTry, numFeatures, random)) {
124       Iterable<Decision> decisions =
125           Decision.decisionsFromExamples(examples, featureNumber, suggestedMaxSplitCandidates);
126       Pair<Decision,Double> decisionAndGain = Information.bestGain(decisions, examples);
127       if (decisionAndGain != null) {
128         double gain = decisionAndGain.getSecond();
129         if (gain > bestGain) {
130           bestGain = gain;
131           bestDecision = decisionAndGain.getFirst();
132         }
133       }
134     }
135 
136     if (Double.isNaN(bestGain) || bestGain < minInfoGainNats) {
137       return new TerminalNode(Prediction.buildPrediction(examples));
138     }
139 
140     bestDecision.setInformationGain(bestGain);
141 
142     ExampleSet[] negPosSplit = examples.split(bestDecision);
143     examples = null; // For GC?
144 
145     TreeNode left = build(negPosSplit[0],
146                           buildAtDepth + 1,
147                           minNodeSize,
148                           minInfoGainNats,
149                           featuresToTry,
150                           numFeatures,
151                           suggestedMaxSplitCandidates,
152                           maxDepth,
153                           random);
154     TreeNode right = build(negPosSplit[1],
155                            buildAtDepth + 1,
156                            minNodeSize,
157                            minInfoGainNats,
158                            featuresToTry,
159                            numFeatures,
160                            suggestedMaxSplitCandidates,
161                            maxDepth,
162                            random);
163 
164     return new DecisionNode(bestDecision, left, right);
165   }

167   private static Iterable<Integer> randomFeatures(ExampleSet examples,
168                                                   int featuresToTry,
169                                                   int numFeatures,
170                                                   RandomGenerator random) {
171     Collection<Integer> features = Sets.newHashSetWithExpectedSize(featuresToTry);
172     int max = FastMath.min(numFeatures, featuresToTry);
173     int attempts = 0;
174     while (features.size() < max && attempts < 2 * featuresToTry) {
175       int featureNumber = random.nextInt(numFeatures);
176       if (examples.getFeatureType(featureNumber) != FeatureType.IGNORED) {
177         features.add(featureNumber);
178       }
179       attempts++;
180     }
181     return features;
182   }

184   @Override
185   public Prediction classify(Example test) {
186     TerminalNode terminalNode = findTerminal(test);
187     return terminalNode.getPrediction();
188   }
189 
190   private TerminalNode findTerminal(Example example) {
191     TreeNode node = root;
192     while (!node.isTerminal()) {
193       DecisionNode decisionNode = (DecisionNode) node;
194       if (decisionNode.getDecision().isPositive(example)) {
195         node = decisionNode.getRight();
196       } else {
197         node = decisionNode.getLeft();
198       }
199     }
200     return (TerminalNode) node;
201   }

203   public double[] featureImportance(ExampleSet testSet) {
204     int numFeatures = testSet.getNumFeatures();
205     double[] totalGains = new double[numFeatures];
206     for (Example test : testSet) {
207       TreeNode node = root;
208       while (!node.isTerminal()) {
209         DecisionNode decisionNode = (DecisionNode) node;
210         Decision decision = decisionNode.getDecision();
211         int feature = decision.getFeatureNumber();
212         totalGains[feature] += decision.getInformationGain();
213         if (decision.isPositive(test)) {
214           node = decisionNode.getRight();
215         } else {
216           node = decisionNode.getLeft();
217         }
218       }
219     }
220     double maxGain = 0.0;
221     for (int i = 0; i < numFeatures; i++) {
222       double gain = totalGains[i];
223       Preconditions.checkState(gain >= 0.0);
224       maxGain = FastMath.max(maxGain, gain);
225     }
226     if (maxGain > 0.0) {
227       for (int i = 0; i < numFeatures; i++) {
228         totalGains[i] /= maxGain;
229       }
230     }
231     // Otherwise all were 0?
232     return totalGains;
233   }

235   @Override
236   public void update(Example train) {
237     TerminalNode terminalNode = findTerminal(train);
238     terminalNode.update(train);
239   }

241   @Override
242   public String toString() {
243     StringBuilder result = new StringBuilder();
244     if (root != null) {
245       Deque<Pair<TreeNode,TreePath>> toPrint = new LinkedList<Pair<TreeNode,TreePath>>();
246       toPrint.push(new Pair<TreeNode,TreePath>(root, TreePath.EMPTY));
247       while (!toPrint.isEmpty()) {
248         Pair<TreeNode,TreePath> entry = toPrint.pop();
249         TreeNode node = entry.getFirst();
250         TreePath path = entry.getSecond();
251         int pathLength = path.length();
252         for (int i = 0; i < pathLength; i++) {
253           if (i == pathLength - 1) {
254             result.append(" +-");
255           } else {
256             result.append(path.isLeftAt(i) ? " | " : "   ");
257           }
258         }
259         result.append(node).append('\n');
260         if (node != null && !node.isTerminal()) {
261           DecisionNode decisionNode = (DecisionNode) node;
262           toPrint.push(new Pair<TreeNode,TreePath>(decisionNode.getRight(), path.extendRight()));
263           toPrint.push(new Pair<TreeNode,TreePath>(decisionNode.getLeft(), path.extendLeft()));
264         }
265       }
266     }
267     return result.toString();
268   }
