16  package com.cloudera.oryx.rdf.common.eval;
17  
18  import java.util.List;
19  
20  import com.cloudera.oryx.common.stats.DoubleWeightedMean;
21  import com.cloudera.oryx.rdf.common.rule.CategoricalPrediction;
22  import com.cloudera.oryx.rdf.common.rule.NumericPrediction;
23  import com.cloudera.oryx.rdf.common.rule.Prediction;
24  
25  /**
26   * A utility class with methods for combining the results of many
27   * {@link com.cloudera.oryx.rdf.common.tree.TreeBasedClassifier}s into one prediction -- in particular, it is used by
28   * {@link com.cloudera.oryx.rdf.common.tree.DecisionForest} to combine the  results of many
29   * {@link com.cloudera.oryx.rdf.common.tree.DecisionTree}s. {@link Prediction}s from many
30   * {@link com.cloudera.oryx.rdf.common.tree.TreeBasedClassifier}s are combined and weighted according to the accuracy of
31   * the individuals.
32   *
33   * @author Sean Owen
34   */
35  public final class WeightedPrediction {
36  
37    private WeightedPrediction() {
38    }
39  
40    /**
41     * @param predictions {@link Prediction}s from individuals
42     * @param weights weights that should be applied to
43     * @return a single {@link Prediction} represented a weighted combination of the inputs
44     */
45    public static <T extends Prediction> Prediction voteOnFeature(List<T> predictions, double[] weights) {
46      switch (predictions.iterator().next().getFeatureType()) {
47        case NUMERIC:
48          @SuppressWarnings("unchecked")
49          List<NumericPrediction> numericVotes = (List<NumericPrediction>) predictions;
50          return voteOnNumericFeature(numericVotes, weights);
51        case CATEGORICAL:
52          @SuppressWarnings("unchecked")
53          List<CategoricalPrediction> categoricalVotes = (List<CategoricalPrediction>) predictions;
54          return voteOnCategoricalFeature(categoricalVotes, weights);
55        default:
56          throw new IllegalStateException();
57      }
58    }
59    
60    private static Prediction voteOnCategoricalFeature(List<CategoricalPrediction> predictions, double[] weights) {
61      float[] weightedProbabilities = null;
62      double totalWeight = 0.0;
63      for (int i = 0; i < predictions.size(); i++) {
64        CategoricalPrediction vote = predictions.get(i);
65        double weight = weights[i];
66        totalWeight += weight;
67        float[] categoryProbabilities = vote.getCategoryProbabilities();
68        if (weightedProbabilities == null) {
69          weightedProbabilities = new float[categoryProbabilities.length];
70        }
71        for (int j = 0; j < weightedProbabilities.length; j++) {
72          weightedProbabilities[j] += categoryProbabilities[j] * weight;
73        }
74      }
75      for (int j = 0; j < weightedProbabilities.length; j++) {
76        weightedProbabilities[j] /= totalWeight;
77      }
78      return new CategoricalPrediction(weightedProbabilities);
79    }
80    
81    private static Prediction voteOnNumericFeature(List<NumericPrediction> predictions, double[] weights) {
82      DoubleWeightedMean mean = new DoubleWeightedMean();
83      for (int i = 0; i < predictions.size(); i++) {
84        mean.increment(predictions.get(i).getPrediction(), weights[i]);
85      }
86      return new NumericPrediction((float) mean.getResult(), (int) mean.getN());
87    }
88  
89  }
