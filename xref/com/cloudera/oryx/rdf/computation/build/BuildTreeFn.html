53  /**
54   * @author Sean Owen
55   */
56  public final class BuildTreeFn extends OryxReduceDoFn<Integer, Iterable<String>, String> {
57  
58    private static final Logger log = LoggerFactory.getLogger(BuildTreeFn.class);
59  
60    private static final CharMatcher NEWLINES = CharMatcher.anyOf("\n\r");
61  
62    private int numLocalTrees;
63    private int trainingFoldsPerTree;
64    private final RandomDataGenerator random = new RandomDataGenerator(RandomManager.getRandom());
65  
66    @Override
67    public void initialize() {
68      super.initialize();
69      int numReducers = getContext().getNumReduceTasks();
70      log.info("{} reducers", numReducers);
71  
72      Config config = ConfigUtils.getDefaultConfig();
73      int numTrees = config.getInt("model.num-trees");
74      // Bump this up to as least 2x reducers
75      numTrees = FastMath.max(numTrees, 2 * numReducers);
76      // Make it a multiple of # reducers
77      while ((numTrees % numReducers) != 0) {
78        numTrees++;
79      }
80      log.info("{} total trees", numTrees);
81  
82      Preconditions.checkArgument(numTrees % numReducers == 0, "numLocalTrees not a multiples of numReducers");
83      numLocalTrees = numTrees / numReducers;
84      Preconditions.checkArgument(numLocalTrees > 1, "Not building multiple trees per reducer");
85      log.info("Building {} trees locally", numLocalTrees);
86  
87      double sampleRate = config.getDouble("model.sample-rate");
88      Preconditions.checkArgument(sampleRate > 0.0 && sampleRate <= 1.0);
89  
90      // Arbitrarily use about 90% for training, locally
91      trainingFoldsPerTree = FastMath.min(numLocalTrees - 1, (int) (0.9 * numLocalTrees));
92  
93      log.info("{} training folds per tree", trainingFoldsPerTree);
94    }
95  
96    @Override
97    public void process(Pair<Integer,Iterable<String>> input, Emitter<String> emitter) {
98  
99      InboundSettings inboundSettings = InboundSettings.create(ConfigUtils.getDefaultConfig());
100     int numFeaturesAndTarget = inboundSettings.getColumnNames().size();
101     Integer targetColumn = inboundSettings.getTargetColumn();
102     Preconditions.checkNotNull(targetColumn, "No target-column specified");
103 
104     List<Example> allExamples = Lists.newArrayList();
105     Map<Integer,BiMap<String,Integer>> columnToCategoryNameToIDMapping = Maps.newHashMap();
106 
107     log.info("Reading input");
108     for (String line : input.second()) {
109       String[] tokens = DelimitedDataUtils.decode(line);
110       Feature target = null;
111       Feature[] features = new Feature[numFeaturesAndTarget]; // Too big by 1 but makes math easier
112       for (int col = 0; col < numFeaturesAndTarget; col++) {
113         if (col == targetColumn) {
114           target = buildFeature(col, tokens[col], inboundSettings, columnToCategoryNameToIDMapping);
115           features[col] = IgnoredFeature.INSTANCE;
116         } else {
117           features[col] = buildFeature(col, tokens[col], inboundSettings, columnToCategoryNameToIDMapping);
118         }
119       }
120       Preconditions.checkNotNull(target);
121       allExamples.add(new Example(target, features));
122     }
123 
124     if (allExamples.isEmpty()) {
125       return;
126     }
127 
128     log.info("Read {} examples", allExamples.size());
129 
130     for (int treeID = 0; treeID < numLocalTrees; treeID++) {
131       List<Example> trainingExamples = Lists.newArrayList();
132       List<Example> cvExamples = Lists.newArrayList();
133       for (Example example : allExamples) {
134         if (random.nextInt(0, numLocalTrees - 1) < trainingFoldsPerTree) {
135           trainingExamples.add(example);
136         } else {
137           cvExamples.add(example);
138         }
139       }
140 
141       log.info("Tree {}: {} training examples", treeID, trainingExamples.size());
142       log.info("Tree {}: {} CV examples", treeID, cvExamples.size());
143       Preconditions.checkState(!trainingExamples.isEmpty(), "No training examples sampled?");
144       Preconditions.checkState(!cvExamples.isEmpty(), "No CV examples sampled?");
145 
146       progress(); // Helps prevent timeouts
147       DecisionTree tree = DecisionTree.fromExamplesWithDefault(trainingExamples);
148       progress(); // Helps prevent timeouts
149       log.info("Built tree {}", treeID);
150       ExampleSet cvSet = new ExampleSet(cvExamples);
151       double[] weightEval = Evaluation.evaluateToWeight(tree, cvSet);
152       double weight = weightEval[0];
153       double eval = weightEval[1];
154       double[] featureImportances = tree.featureImportance(cvSet);
155       progress(); // Helps prevent timeouts
156       log.info("Evaluated tree {}: {}", treeID, eval);
157       log.info("Feature importances: {}", featureImportances);
158 
159       DecisionForest singletonForest = new DecisionForest(
160           new DecisionTree[] { tree }, new double[] { weight }, featureImportances);
161 
162       String pmmlFileContents;
163       try  {
164         StringWriter treePMML = new StringWriter();
165         DecisionForestPMML.write(treePMML, singletonForest, columnToCategoryNameToIDMapping);
166         pmmlFileContents = NEWLINES.removeFrom(treePMML.toString());
167       } catch (IOException ioe) {
168         throw new IllegalStateException(ioe);
169       }
170 
171       log.info("Emitting tree {}", treeID);
172       emitter.emit(pmmlFileContents);
173     }
174 
175   }
176 
177   private static Feature buildFeature(int columnNumber,
178                                       String token,
179                                       InboundSettings inboundSettings,
180                                       Map<Integer, BiMap<String, Integer>> columnToCategoryNameToIDMapping) {
181     if (token == null || token.isEmpty()) {
182       return null;
183     }
184     if (inboundSettings.isNumeric(columnNumber)) {
185       return NumericFeature.forValue(Float.parseFloat(token));
186     }
187     if (inboundSettings.isCategorical(columnNumber)) {
188       return CategoricalFeature.forValue(categoricalFromString(columnNumber, token, columnToCategoryNameToIDMapping));
189     }
190     return IgnoredFeature.INSTANCE;
191   }
192 
193   private static int categoricalFromString(int columnNumber,
194                                            String value,
195                                            Map<Integer, BiMap<String, Integer>> columnToCategoryNameToIDMapping) {
196     BiMap<String,Integer> categoryNameToID = columnToCategoryNameToIDMapping.get(columnNumber);
197     if (categoryNameToID == null) {
198       categoryNameToID = HashBiMap.create();
199       columnToCategoryNameToIDMapping.put(columnNumber, categoryNameToID);
200     }
201     Integer mapped = categoryNameToID.get(value);
202     if (mapped != null) {
203       return mapped;
204     }
205     int newCategory = categoryNameToID.size();
206     categoryNameToID.put(value, newCategory);
207     return newCategory;
208   }
