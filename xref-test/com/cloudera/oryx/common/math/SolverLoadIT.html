31  /**
32   * Tests performance of {@link LinearSystemSolver#getSolver(RealMatrix)}.
33   * 
34   * @author Sean Owen
35   */
36  public final class SolverLoadIT extends OryxTest {
37    
38    private static final Logger log = LoggerFactory.getLogger(SolverLoadIT.class);
39    
40    @Test
41    public void testLoad() {
42      RealMatrix symmetric = randomSymmetricMatrix(500);
43      Stopwatch stopwatch = new Stopwatch().start();
44      int iterations = 100;
45      for (int i = 0; i < iterations; i++) {
46        MatrixUtils.getSolver(symmetric);
47      }
48      stopwatch.stop();
49      // long elapsedMS = stopwatch.elapsed(TimeUnit.MILLISECONDS);
50      long elapsedMS = stopwatch.elapsedTime(TimeUnit.MILLISECONDS);
51      log.info("{}ms elapsed for {} iterations", elapsedMS, iterations);
52      assertTrue(elapsedMS < 800 * iterations);
53    }
54    
55    private static RealMatrix randomSymmetricMatrix(int dimension) {
56      RandomGenerator random = RandomManager.getRandom();
57      RealMatrix symmetric = new Array2DRowRealMatrix(dimension, dimension);
58      for (int j = 0; j < dimension; j++) {
59        // Diagonal
60        symmetric.setEntry(j, j, random.nextDouble());
61        for (int k = j + 1; k < dimension; k++) {
62          // Off-diagonal
63          double d = random.nextDouble();
64          symmetric.setEntry(j, k, d);
65          symmetric.setEntry(k, j, d);          
66        }
67      }
68      return symmetric;
69    }
