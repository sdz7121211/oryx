44  public final class SummaryTest extends OryxTest {
45  
46    private static final PCollection<RealVector> VECS = MemPipeline.typedCollectionOf(
47        MLAvros.vector(),
48        Vectors.of(1.0, 3.0),
49        Vectors.of(1.0, 1.0),
50        Vectors.of(3.0, 1.0),
51        Vectors.of(3.0, 3.0));
52    
53    private static final PCollection<String> STRINGS = MemPipeline.typedCollectionOf(
54        Avros.strings(),
55        "1.0,NA,2.0",
56        "1.0,2.0,3.0");
57  
58    @Test
59    public void testZScores() {
60      PCollection<Record> elems = VECS.parallelDo(new MapFn<RealVector, Record>() {
61        @Override
62        public Record map(RealVector vec) {
63          return new VectorRecord(vec);
64        }
65      }, null);
66      Summarizer sr = new Summarizer();
67      Summary s = sr.build(elems).getValue();
68      StandardizeFn fn = new StandardizeFn(s, Transform.Z);
69      assertEquals(ImmutableList.of(Vectors.of(-1, 1),
70          Vectors.of(-1, -1), Vectors.of(1, -1),
71          Vectors.of(1, 1)), elems.parallelDo(fn, MLAvros.vector()).materialize());
72    }
73    
74    @Test
75    public void testMissing() throws Exception {
76      PCollection<Record> elems = STRINGS.parallelDo(new MapFn<String, Record>() {
77        @Override
78        public Record map(String input) {
79          return new CSVRecord(Arrays.asList(input.split(",")));
80        }
81      }, MLRecords.csvRecord(AvroTypeFamily.getInstance(), ","));
82      Summarizer sr = new Summarizer();
83      Summary s = sr.build(elems).getValue();
84      assertEquals(1, s.getStats(1).getMissing());
85      assertEquals(2.0, s.getStats(1).mean(), 0.01);
86      assertEquals(0.0, s.getStats(1).stdDev(), 0.01);
87    }
88    
89    @Test
90    public void testTrailingIgnoredFields() throws Exception {
91      Spec spec = RecordSpec.builder().add("field1", DataType.DOUBLE)
92          .add("field2", DataType.DOUBLE).add("field3", DataType.DOUBLE).build();
93      PCollection<Record> elems = STRINGS.parallelDo(new MapFn<String, Record>() {
94        @Override
95        public Record map(String input) {
96          return new CSVRecord(Arrays.asList(input.split(",")));
97        }
98      }, MLRecords.csvRecord(AvroTypeFamily.getInstance(), ","));
99      Summarizer sr = new Summarizer().spec(spec).ignoreColumns(2);
100     sr.build(elems).getValue();
101   }
102 }
