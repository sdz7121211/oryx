16  package com.cloudera.oryx.kmeans.computation.cluster;
17  
18  import java.util.List;
19  
20  import com.cloudera.oryx.common.OryxTest;
21  import com.cloudera.oryx.common.random.RandomManager;
22  import com.cloudera.oryx.kmeans.common.Centers;
23  
24  import com.cloudera.oryx.kmeans.computation.MLAvros;
25  import org.apache.commons.math3.linear.RealVector;
26  import org.apache.crunch.PCollection;
27  import org.apache.crunch.impl.mem.MemPipeline;
28  import org.junit.Before;
29  import org.junit.Test;
30  
31  import com.cloudera.oryx.common.math.Vectors;
32  import com.google.common.collect.ImmutableList;
33  
34  public final class KMeansParallelTest extends OryxTest {
35    
36    private static final PCollection<RealVector> VECS = MemPipeline.typedCollectionOf(
37        MLAvros.vector(),
38        Vectors.of(2.0, 1.0),
39        Vectors.of(1.0, 1.0),
40        Vectors.of(5.0, 4.0),
41        Vectors.of(4.0, 3.0),
42        Vectors.of(2.0, 1.0),
43        Vectors.of(1.0, 1.0),
44        Vectors.of(5.0, 4.0),
45        Vectors.of(4.0, 3.0),
46        Vectors.of(2.0, 1.0),
47        Vectors.of(1.0, 1.0),
48        Vectors.of(5.0, 4.0),
49        Vectors.of(4.0, 3.0),
50        Vectors.of(2.0, 1.0),
51        Vectors.of(1.0, 1.0),
52        Vectors.of(5.0, 4.0),
53        Vectors.of(4.0, 3.0));
54    
55    private KMeansParallel kmp;
56  
57    @Override
58    @Before
59    public void setUp() throws Exception {
60      super.setUp();
61      kmp = new KMeansParallel(RandomManager.getRandom(), 128, 32);
62    }
63    
64    @Test
65    public void testLloyds() throws Exception {
66      List<Centers> centers = ImmutableList.of(
67          new Centers(ImmutableList.of(Vectors.of(1.0, 1.0), Vectors.of(5.0, 4.0))));
68      List<Centers> res = kmp.lloydsAlgorithm(VECS, centers, 0, false);
69      assertEquals(centers, res);
70      
71      res = kmp.lloydsAlgorithm(VECS, res, 1, false);
72      List<Centers> expected = ImmutableList.of(
73          new Centers(ImmutableList.of(Vectors.of(1.5, 1.0), Vectors.of(4.5, 3.5))));
74      assertEquals(expected, res);
75      
76      res = kmp.lloydsAlgorithm(VECS, res, 1, false);
77      assertEquals(expected, res);
78    }
79  }
