29  /**
30   * Tests {@link AlternatingLeastSquares} with negative input values.
31   *
32   * @author Sean Owen
33   */
34  public final class NegativeInputTest extends OryxTest {
35  
36    @Test
37    public void testALS() throws Exception {
38  
39      LongObjectMap<LongFloatMap> byRow = new LongObjectMap<LongFloatMap>();
40      LongObjectMap<LongFloatMap> byCol = new LongObjectMap<LongFloatMap>();
41  
42      // Octave: R = [ 1 1 1 0 ; 0 -1 1 1 ; -1 0 0 1 ]
43      MatrixUtils.addTo(0, 0,  1.0f, byRow, byCol);
44      MatrixUtils.addTo(0, 1,  1.0f, byRow, byCol);
45      MatrixUtils.addTo(0, 2,  1.0f, byRow, byCol);
46      MatrixUtils.addTo(1, 1, -1.0f, byRow, byCol);
47      MatrixUtils.addTo(1, 2,  1.0f, byRow, byCol);
48      MatrixUtils.addTo(1, 3,  1.0f, byRow, byCol);
49      MatrixUtils.addTo(2, 0, -1.0f, byRow, byCol);
50      MatrixUtils.addTo(2, 3,  1.0f, byRow, byCol);
51  
52      // Octave: Y = [ 0.1 0.2 ; 0.2 0.5 ; 0.3 0.1 ; 0.2 0.2 ];
53      LongObjectMap<float[]> previousY = new LongObjectMap<float[]>();
54      previousY.put(0L, new float[] {0.1f, 0.2f});
55      previousY.put(1L, new float[] {0.2f, 0.5f});
56      previousY.put(2L, new float[] {0.3f, 0.1f});
57      previousY.put(3L, new float[] {0.2f, 0.2f});
58  
59      MatrixFactorizer als = new AlternatingLeastSquares(byRow, byCol, 2, 0.0001, 40);
60      als.setPreviousY(previousY);
61      als.call();
62  
63      RealMatrix product = MatrixUtilsTest.multiplyXYT(als.getX(), als.getY());
64  
65      assertArrayEquals(
66          new float[] {0.899032f, 0.900162f, 0.990150f, -0.026642f},
67          product.getRow(0));
68      assertArrayEquals(
69          new float[] {0.181214f, 0.089988f, 0.787198f, 1.012226f},
70          product.getRow(1));
71      assertArrayEquals(
72          new float[] {-0.104165f, -0.178240f, 0.360391f, 0.825856f},
73          product.getRow(2));
74    }
75  
76  }
