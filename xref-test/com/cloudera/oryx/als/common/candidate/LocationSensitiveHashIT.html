41  /**
42   * A long-runnign integration test for {@link LocationSensitiveHash}.
43   *
44   * @author Sean Owen
45   */
46  public final class LocationSensitiveHashIT extends OryxTest {
47  
48    private static final Logger log = LoggerFactory.getLogger(LocationSensitiveHashIT.class);
49  
50    private static final int NUM_FEATURES = 30;
51    private static final int NUM_ITEMS = 1000000;
52    private static final int NUM_RECS = 10;
53    private static final int ITERATIONS = 10;
54    private static final double LN2 = Math.log(2.0);
55  
56    @Test
57    public void testLSH() {
58      RandomGenerator random = RandomManager.getRandom();
59  
60      Mean avgPercentTopRecsConsidered = new Mean();
61      Mean avgNDCG = new Mean();
62      Mean avgPercentAllItemsConsidered = new Mean();
63  
64      for (int iteration = 0; iteration < ITERATIONS; iteration++) {
65  
66        LongObjectMap<float[]> Y = new LongObjectMap<float[]>();
67        for (int i = 0; i < NUM_ITEMS; i++) {
68          Y.put(i, RandomUtils.randomUnitVector(NUM_FEATURES, random));
69        }
70        float[] userVec = RandomUtils.randomUnitVector(NUM_FEATURES, random);
71  
72        double[] results = doTestRandomVecs(Y, userVec);
73        double percentTopRecsConsidered = results[0];
74        double ndcg = results[1];
75        double percentAllItemsConsidered = results[2];
76  
77        log.info("Considered {}% of all candidates, {} nDCG, got {}% recommendations correct",
78                 100 * percentAllItemsConsidered,
79                 ndcg,
80                 100 * percentTopRecsConsidered);
81  
82        avgPercentTopRecsConsidered.increment(percentTopRecsConsidered);
83        avgNDCG.increment(ndcg);
84        avgPercentAllItemsConsidered.increment(percentAllItemsConsidered);
85      }
86  
87      log.info("{}", avgPercentTopRecsConsidered.getResult());
88      log.info("{}", avgNDCG.getResult());
89      log.info("{}", avgPercentAllItemsConsidered.getResult());
90  
91      assertTrue(avgPercentTopRecsConsidered.getResult() > 0.8);
92      assertTrue(avgNDCG.getResult() > 0.8);
93      assertTrue(avgPercentAllItemsConsidered.getResult() < 0.09);
94    }
95  
96    private static double[] doTestRandomVecs(LongObjectMap<float[]> Y, float[] userVec) {
97  
98      LocationSensitiveHash lsh = new LocationSensitiveHash(Y, 0.1, 20);
99  
100     LongSet candidates = new LongSet();
101     float[][] userVecs = { userVec };
102     for (Iterator<LongObjectMap.MapEntry<float[]>> candidatesIterator : lsh.getCandidateIterator(userVecs)) {
103       while (candidatesIterator.hasNext()) {
104         candidates.add(candidatesIterator.next().getKey());
105       }
106     }
107 
108     List<Long> topIDs = findTopRecommendations(Y, userVec);
109 
110     double score = 0.0;
111     double maxScore = 0.0;
112     int intersectionSize = 0;
113     for (int i = 0; i < topIDs.size(); i++) {
114       double value = LN2 / Math.log(2.0 + i);
115       long id = topIDs.get(i);
116       if (candidates.contains(id)) {
117         intersectionSize++;
118         score += value;
119       }
120       maxScore += value;
121     }
122 
123     double percentTopRecsConsidered = (double) intersectionSize / topIDs.size();
124     double ndcg = maxScore == 0.0 ? 0.0 : score / maxScore;
125     double percentAllItemsConsidered = (double) candidates.size() / Y.size();
126 
127     return new double[] {percentTopRecsConsidered, ndcg, percentAllItemsConsidered};
128   }
129 
130   private static List<Long> findTopRecommendations(LongObjectMap<float[]> Y, float[] userVec) {
131     // SortedMap<Double,Long> allScores = Maps.newTreeMap(Collections.reverseOrder());
132     // Above triggers some weird OpenJDK 1.6.0_30 compiler bug. Use equivalent:
133     SortedMap<Double,Long> allScores = new TreeMap<Double,Long>(Collections.reverseOrder());
134     for (LongObjectMap.MapEntry<float[]> entry : Y.entrySet()) {
135       double dot = SimpleVectorMath.dot(entry.getValue(), userVec);
136       allScores.put(dot, entry.getKey());
137     }
138     List<Long> topRecommendations = Lists.newArrayList();
139     for (Map.Entry<Double,Long> entry : allScores.entrySet()) {
140       topRecommendations.add(entry.getValue());
141       if (topRecommendations.size() == NUM_RECS) {
142         return topRecommendations;
143       }
144     }
145     return topRecommendations;
146   }
