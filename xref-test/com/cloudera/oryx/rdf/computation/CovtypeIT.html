39  /**
40   * <p>Tests the random decision forest classifier using the
41   * <a href="http://archive.ics.uci.edu/ml/datasets/Covertype">Covtype data set</a>, which contains
42   * both numeric and categorical features and a categorical target.</p>
43   *
44   * <p>Covtype data set is copyright Jock A. Blackard and Colorado State University.</p>
45   *
46   * @author Sean Owen
47   */
48  public final class CovtypeIT extends AbstractComputationIT {
49  
50    private static final Logger log = LoggerFactory.getLogger(CovtypeIT.class);
51  
52    @Override
53    protected File getTestDataPath() {
54      return getResourceAsFile("covtype");
55    }
56  
57    @Test
58    public void testCovtype() throws Exception {
59      List<Example> allExamples = readCovtypeExamples();
60      DecisionForest forest = DecisionForest.fromExamplesWithDefault(allExamples);
61      log.info("Evals: {}", forest.getEvaluations());
62      assertTrue(new Mean().evaluate(forest.getEvaluations()) >= 0.8);
63      double[] importances = forest.getFeatureImportances();
64      log.info("Importances: {}", importances);
65      assertNotNull(importances);
66      for (double d : importances) {
67        assertTrue(d >= 0.0);
68        assertTrue(d <= 1.0);
69      }
70      assertEquals(importances[0], Doubles.max(importances));
71      // Assert something about important features
72      assertTrue(importances[0] > 0.9);
73      assertTrue(importances[5] > 0.4);
74      assertTrue(importances[9] > 0.4);
75      assertTrue(importances[13] > 0.4);
76    }
77  
78    @Test
79    public void testPMMLOutput() throws Exception {
80      new RDFLocalGenerationRunner().call();
81      File pmmlFile = new File(TEST_TEMP_BASE_DIR, "00000/model.pmml.gz");
82      DecisionForestPMML.read(pmmlFile);
83    }
84  
85    private static List<Example> readCovtypeExamples() throws IOException {
86      List<Example> allExamples = Lists.newArrayList();
87      Pattern delimiter = Pattern.compile(",");
88      File dataFile = new File(TEST_TEMP_INBOUND_DIR, "covtype.csv.gz");
89      for (CharSequence line : new FileLineIterable(dataFile)) {
90        String[] tokens = delimiter.split(line);
91        Feature[] features = new Feature[54];
92        for (int i = 0; i < 10; i++) {
93          features[i] = NumericFeature.forValue(Float.parseFloat(tokens[i]));
94        }
95        for (int i = 10; i < 54; i++) {
96          features[i] = CategoricalFeature.forValue(Integer.parseInt(tokens[i]));
97        }
98        Example trainingExample = new Example(CategoricalFeature.forValue(Integer.parseInt(tokens[54]) - 1), features);
99        allExamples.add(trainingExample);
100     }
101     return allExamples;
102   }
