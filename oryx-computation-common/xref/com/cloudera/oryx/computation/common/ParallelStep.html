32  import org.slf4j.Logger;
33  import org.slf4j.LoggerFactory;

38  /**
39   * A step which actually runs several {@link JobStep}s in parallel.
40   *
41   * @author Sean Owen
42   */
43  public final class ParallelStep extends Configured implements Tool, HasState {
44  
45    private static final Logger log = LoggerFactory.getLogger(ParallelStep.class);
46  
47    private final Collection<JobStep> steps;
48  
49    public ParallelStep() {
50      // steps = Lists.newCopyOnWriteArrayList();
51      steps = new CopyOnWriteArrayList<JobStep>();
52    }
53  
54    /**
55     * @return result of {@code getStepStates()} from this step's parallel component steps.
56     */
57    @Override
58    public Collection<StepState> getStepStates() throws IOException, InterruptedException {
59      Collection<StepState> states = Lists.newArrayList();
60      for (JobStep step : steps) {
61        states.addAll(step.getStepStates());
62      }
63      return states;
64    }
65  
66    /**
67     * @param args will be passed through to the parallel steps, except for the last. The last is a
68     * comma-separated list of class names of steps to execute in parallel.
69     */
70    @Override
71    public int run(final String[] args) throws JobException, IOException, InterruptedException {
72  
73      // all but last arg are passed through
74      String[] stepClassNames = args[args.length - 1].split(",");
75  
76      Collection<Callable<Object>> runnables = Lists.newArrayList();
77  
78      for (String stepClassName : stepClassNames) {
79        final JobStep step = ClassUtils.loadInstanceOf(stepClassName, JobStep.class);
80        steps.add(step);
81        runnables.add(new Callable<Object>() {
82          @Override
83          public Void call() throws Exception {
84            log.info("Running step {}", step.getClass().getSimpleName());
85            ToolRunner.run(getConf(), step, args);
86            return null;
87          }
88        });
89      }
90  
91      ExecutorService executor = Executors.newCachedThreadPool(
92          new ThreadFactoryBuilder().setNameFormat("ParallelStep-%d").build());
93      Iterable<Future<Object>> futures;
94      try {
95        futures = executor.invokeAll(runnables);
96      } finally {
97        // shutdown() now rather than shutdownNow() later; this may let some parallel jobs finish
98        executor.shutdown();
99      }
100     for (Future<?> future : futures) {
101       try {
102         future.get();
103       } catch (ExecutionException e) {
104         Throwable cause = e.getCause();
105         if (cause instanceof InterruptedException) {
106           log.warn("Interrupted");
107           throw (InterruptedException) cause;
108         }
109         if (cause instanceof IOException) {
110           log.warn("Unexpected exception while running step", cause);
111           throw (IOException) cause;
112         }
113         if (cause instanceof JobException) {
114           log.warn("Unexpected exception while running step", cause);
115           throw (JobException) cause;
116         }
117         log.error("Unexpected exception while running step", cause);
118         throw new JobException(cause);
119       }
120     }
121 
122     return 0;
123   }
124 
125   public static void main(String[] args) throws Exception {
126     ToolRunner.run(new OryxConfiguration(), new ParallelStep(), args);
127   }
