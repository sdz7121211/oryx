335  private static boolean isUploadInProgress(String uploadingGenerationPrefix) throws IOException {
336    Store store = Store.get();
337    long now = System.currentTimeMillis();
338    for (String fileName : store.list(uploadingGenerationPrefix, true)) {
339      long lastModified = store.getLastModified(fileName);
340 
341      if (fileName.endsWith(".inprogress") || fileName.endsWith("_COPYING_")) {
342        // .inprogress is our marker for uploading files; _COPYING_ is Hadoop's
343        if (lastModified > now - 3 * GENERATION_WAIT) {
344          log.info("At least one upload is in progress ({})", fileName);
345          return true;
346        }
347        // Else, orphaned upload -- waited a long, long time with no progress
348        log.warn("Stale upload to {}? Deleting and continuing", fileName);
349        try {
350          store.delete(fileName);
351        } catch (IOException ioe) {
352          log.info("Could not delete {}", fileName, ioe);
353        }
354      } else {
355        // All other (data) files. Look for recent modification, but only quite recent.
356        // This accounts for possible side-loads of data from other processes.
357        if (lastModified > now - GENERATION_WAIT) {
358          log.info("At least one upload is in progress ({})", fileName);
359          return true;
360        }
361      }
362    }
363    return false;
364  }
